60 static std::unique_ptr<Postlist<value_type> > merge(std::vector<FILE*>& buckets) {
61 // initialize value streams
62 std::vector<value_type> values(buckets.size());
63 for (unsigned i(0); i != buckets.size(); ++i) {
64 std::rewind(buckets[i]);
65 if (!traits_type::read_from(values[i], buckets[i])) {
66 throw std::runtime_error(std::format("traits_type::read_from failed for : {}", values[i]));
67 }
68 }
69 // merge value streams
70 size_t bucket_index;
71 PostlistBuilder<value_type> builder;
72 typename std::vector<value_type>::iterator next_value;
73 while (!values.empty()) {
74 next_value = std::min_element(values.begin(), values.end(), compare_type());
75 builder.push_back(*next_value);
76 bucket_index = std::distance(values.begin(), next_value);
77 if (!traits_type::read_from(*next_value, buckets[bucket_index])) {
78 values.erase(next_value);
79 util::fclose(buckets[bucket_index]);
80 buckets.erase(buckets.begin() + bucket_index);
81 }
82 }
83 return builder.build();
84 }

86 static const std::vector<FILE*> scatter(Postlist<value_type>& postlist, size_t bucket_count) {
87 value_type value;
88 postlist.rewind();
89 std::vector<FILE*> bucket_fs;
90 std::vector<value_type> values;
91 if (bucket_count == 0)
92 ++bucket_count;
93 const size_t max_values_per_bucket(1 + postlist.size() / bucket_count);
94 values.reserve(max_values_per_bucket);
95 while (postlist.next(value)) {
96 values.push_back(value);
97 if (values.size() == max_values_per_bucket) {
98 std::sort(values.begin(), values.end(), compare_type());
99 bucket_fs.push_back(write_to_tmpfile(values));
100 values.clear();
101 }
102 }
103 if (!values.empty()) {
104 std::sort(values.begin(), values.end(), compare_type());
105 bucket_fs.push_back(write_to_tmpfile(values));
106 }
107 return bucket_fs;
108 }
