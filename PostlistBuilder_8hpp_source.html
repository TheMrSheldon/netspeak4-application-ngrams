39  std::unique_ptr<Postlist<value_type> > build() {
40  if (sizes_.empty()) {
41  return std::unique_ptr<Postlist<value_type> >(new Postlist<value_type>);
42  }
43 
44  head_.value_count = sizes_.size();
45  head_.value_size = sizes_.front();
46  head_.total_size = std::accumulate(sizes_.begin(), sizes_.end(), 0);
47  for (auto it(sizes_.begin()); it != sizes_.end(); ++it) {
48  if (*it != head_.value_size) {
49  head_.value_size = 0; // indicates variable value size
50  break;
51  }
52  }
53 
54  std::unique_ptr<Postlist<value_type> > postlist;
55  if (has_swap_()) {
56  util::rewind(swap_.stream_);
57  swap_.pagesize_ = page_size_;
58  postlist.reset((head_.value_size == 0) ? new Postlist<value_type>(head_, swap_, sizes_)
59  : new Postlist<value_type>(head_, swap_));
60  swap_.stream_ = nullptr; // postlist gets file stream ownership
61  } else {
62  page_type page(head_.total_size);
63  for (auto it(values_.begin()); it != values_.end(); ++it) {
64  if (page.buffer_.put(*it))
65  continue;
66  throw std::runtime_error(std::format("ByteBuffer::put failed : {}", *it));
67  }
68  assert(page.buffer_.position() == page.buffer_.end());
69  postlist.reset((head_.value_size == 0) ? new Postlist<value_type>(head_, page, sizes_)
70  : new Postlist<value_type>(head_, page));
71  }
72  clear();
73  return postlist;
74  }
75 
76  void clear() {
77  util::fclose(swap_.stream_);
78  head_ = Head();
79  swap_ = swap_type();
80  values_.clear();
81  sizes_.clear();
82  mem_usage_ = 0;
83  }

85  void push_back(const value_type& value) {
86  sizes_.push_back(traits_type::size_of(value));
87  if (has_swap_()) {
88  traits_type::write_to(value, swap_.stream_);
89  } else {
90  values_.push_back(value);
91  mem_usage_ += sizes_.back();
92  if (mem_usage_ > page_size_) {
93  swap_out_values_();
94  }
95  }
96  }
97 
98 private:
99  void swap_out_values_() {
100  swap_.stream_ = util::tmpfile();
101  for (auto it(values_.begin()); it != values_.end(); ++it) {
102  traits_type::write_to(*it, swap_.stream_);
103  }
104  values_.clear();
105  mem_usage_ = 0;
106  }
107 
108  bool has_swap_() const {
109  return swap_.stream_ != nullptr;
110  }
