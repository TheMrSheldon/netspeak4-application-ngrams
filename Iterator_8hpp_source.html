150 inline size_t swap() {
151 const size_t remaining_value_count(size() - page_.index_end_);
152 const size_t max_values_per_page(swap_.pagesize_ / size_);
153 const size_t new_value_count(std::min(remaining_value_count, max_values_per_page));
154 if (new_value_count != 0) {

156 page_.buffer_.resize(new_buffer_size);

158 page_.index_begin_ = page_.index_end_;
159 page_.index_cur_ = page_.index_begin_;
160 page_.index_end_ += new_value_count;
