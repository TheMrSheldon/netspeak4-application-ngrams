101 if (!result.unknown_words().empty()) {
102 std::cout << " (" << result.unknown_words().size() << " unknown word(s):";

107 }

185void handle_corpus_key(std::string& corpus_key, netspeak::service::NetspeakService::Stub& stub) {
186 service::CorporaResponse corporaRes = getCorpora(stub);
187 const auto& corpora = corporaRes.corpora();
188 if (corpus_key.empty()) {
189 // select key
190
191 if (corpora.empty()) {
192 throw std::logic_error("The server doesn't support any corpora.");
193 } else if (corpora.size() == 1) {
194 const auto& corpus = *corpora.begin();
195 std::cout << "The corpus " << corpus << " has been selected because it's the only corpus.\n";
196 corpus_key = corpus.key();
197 } else {
198 std::cout << "The server supports multiple corpora. Please choose one by "
199 "entering its number:\n";
200 int i = 0;
201 for (const auto& corpus : corpora) {
202 std::cout << "\t" << (i++) << ": " << corpus << "\n";
203 }
204 int index;
205 std::cin >> index;
206 if (index < 0 || index >= corpora.size()) {
207 throw std::logic_error("Number out of range.");
208 }
209
210 const auto& corpus = corporaRes.corpora(index);
211 std::cout << "The corpus " << corpus << " has been selected.\n";
212 corpus_key = corpus.key();
213 }
214 } else {
215 // verify key
216
217 if (std::find_if(corpora.begin(), corpora.end(),
218 [&](const service::Corpus& corpus) { return corpus.key() == corpus_key; }) == corpora.end()) {
219 std::stringstream what;
220 what << "Cannot find the given corpus key \"" << corpus_key
221 << "\" in the list of corpora support by the server.\n"
222 << "Supported are:";
223 for (const auto& corpus : corpora) {
224 what << "\n\t" << corpus;

226 throw std::logic_error(what.str());

230void RunLocalServerShell(const std::string& address, std::string corpus_key) {
231 auto channel = grpc::CreateChannel(address, grpc::InsecureChannelCredentials());
232 auto stub = service::NetspeakService::NewStub(channel);
233
234 handle_corpus_key(corpus_key, *stub);
235
236 Searcher searcher = [&](const service::SearchRequest& request, service::SearchResponse& response) {
237 service::SearchRequest req(request);
238 req.set_corpus(corpus_key);
239
240 grpc::ClientContext context;
241 auto status = stub->Search(&context, req, &response);
242 if (!status.ok()) {
243 std::stringstream what;
244 what << status;
245 throw std::logic_error(what.str());
246 }
247 };
248 RunShell(searcher);
