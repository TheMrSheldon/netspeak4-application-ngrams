66 // How does this class work?
67 //
68 // Internally, we represent each unit of the chain with a 0 and each cut
69 // with a 1 in an array. Our state array will then contain length-many 0s
70 // and cuts-many 1s. If we then go through all unique permutations of the
71 // state array, we get exactly what we want. (Just look at the above state
72 // diagram for the example. Each state disgram is exactly the internal state
73 // used to derive the output.)
74
75 // initialize states

80 state_.push_back(1);
81 }
82
83 // initialize output
84 output_.reserve(cuts + 1);

86 output_.push_back(0);
