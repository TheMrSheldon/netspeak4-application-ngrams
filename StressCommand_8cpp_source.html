147 static time_analysis analyse(const std::vector<std::chrono::nanoseconds>& durations) {
148 std::chrono::nanoseconds min(INT64_MAX);
149 std::chrono::nanoseconds max(0);
150 double avg = 0;
151
152 for (const auto& d : durations) {
153 min = std::min(min, d);
154 max = std::max(max, d);
155 avg += d.count();
156 }
157 avg /= durations.size();
158
159 double var = 0;
160
161 for (const auto& d : durations) {
162 auto diff = d.count() - avg;
163 var += diff * diff;
164 }
165 var /= durations.size();
166
167 return time_analysis(time_analysis::sec(avg * 1e-9), time_analysis::sec(std::sqrt(var) * 1e-9),
168 std::chrono::duration_cast<time_analysis::sec>(max),
169 std::chrono::duration_cast<time_analysis::sec>(min));
170 }

181 void start(const uint32_t concurrency, const uint64_t max_queries) {
182 const auto corpus_keys = get_corpus_keys();
183 if (corpus_keys.empty()) {
184 throw std::logic_error("The Netspeak server has to have at least one corpus.");
185 }
186
187 std::vector<std::unique_ptr<corpus_measurements>> measurements_list;
188 for (auto i = corpus_keys.size(); i != 0; i--) {
189 measurements_list.push_back(std::make_unique<corpus_measurements>());
190 }
191
192 std::atomic<uint64_t> req_counter(0);
193
194 std::vector<std::unique_ptr<std::thread>> threads;
195 for (auto i = concurrency; i != 0; i--) {
196 // worker threads
197 threads.push_back(std::make_unique<std::thread>([&]() {
198 // setup the stub
199 auto channel = grpc::CreateChannel(address_, grpc::InsecureChannelCredentials());
200 auto stub = service::NetspeakService::NewStub(channel);
201
202 // setup random
203 std::default_random_engine rnd;
204 std::uniform_int_distribution<int> keys_d(0, corpus_keys.size() - 1);
205 std::uniform_int_distribution<int> queries_d(0, queries_.size() - 1);
206
207 while (++req_counter <= max_queries) {
208 grpc::ClientContext context;
209 service::SearchRequest req;
210 service::SearchResponse resp;
211
212 service::set_tracking_id(context, tracking_id_);
213
214 const auto query_index = queries_d(rnd);
215 const auto corpus_keys_index = keys_d(rnd);
216 auto& measurements = *measurements_list[corpus_keys_index];
217
218 req.set_query(queries_[query_index]);
219 req.set_corpus(corpus_keys[corpus_keys_index]);
220
221 const auto start = std::chrono::steady_clock::now();
222 const auto status = safe_search(*stub, &context, req, &resp);
223
224 measurements.add_duration(std::chrono::steady_clock::now() - start);
225 measurements.req_counter_++;
226
227 if (!status.ok()) {
228 measurements.grpc_error_counter_++;
229 } else if (resp.has_error()) {
230 measurements.resp_error_counter_++;
231 } else {
232 measurements.success_counter_++;
233 }
234 }
235 }));
236 }
237
238 // display output
239 uint64_t last_req_counter = 0;
240 const auto display_start = std::chrono::steady_clock::now();
241 while (req_counter <= max_queries) {
242 std::this_thread::sleep_for(std::chrono::seconds(1));
243
244 uint64_t curr_req_counter = req_counter;
245 uint64_t diff_req_counter = curr_req_counter - last_req_counter;
246 last_req_counter = curr_req_counter;
247
248 // lock all measurements
249 std::vector<std::unique_ptr<std::lock_guard<std::mutex>>> locks;
250 for (auto& m : measurements_list) {
251 locks.push_back(std::make_unique<std::lock_guard<std::mutex>>(m->mutex_));
252 }
253
254 auto seconds_since_start =
255 std::chrono::duration_cast<std::chrono::duration<double>>(std::chrono::steady_clock::now() - display_start)
256 .count();
257 auto avg_req_per_s = curr_req_counter / seconds_since_start;
258 auto curr_req_per_s = diff_req_counter;
259
260 std::cout << std::fixed << std::setprecision(2) << "Average: " << avg_req_per_s << "req/s"
261 << "\tCurrent: " << curr_req_per_s << "req" << std::endl;
262
263 size_t key_pad = 0;
264 for (const auto& key : corpus_keys) {
265 key_pad = std::max(key_pad, key.size());
266 }
267 key_pad += 2;
268
269 for (auto i = 0; i < corpus_keys.size(); i++) {
270 const auto& key = corpus_keys[i];
271 const auto& m = *measurements_list[i];
272
273 auto total_req = static_cast<double>(m.req_counter_);
274 auto success_percentage = m.success_counter_ / total_req * 100.;
275 auto grpc_err_percentage = m.grpc_error_counter_ / total_req * 100.;
276 auto resp_err_percentage = m.resp_error_counter_ / total_req * 100.;
277
278 std::cout << " " << key << std::string(key_pad - key.size(), ' ') << " Succ: " << success_percentage << "%"
279 << " Err: " << resp_err_percentage << "%"
280 << " gRPC Err: " << grpc_err_percentage << "%" << std::endl;
281
282 auto dur_analysis = analyse(m.durations_);
283
284 std::cout << " " << std::string(key_pad, ' ') << " avg: " << get_ms(dur_analysis.avg_) << "ms"
285 << " std: " << get_ms(dur_analysis.std_) << "ms"
286 << " min: " << get_ms(dur_analysis.min_) << "ms"
287 << " max: " << get_ms(dur_analysis.max_) << "ms" << std::endl;
288 }
289 }
290
291 // wait for all threads to die
292 for (const auto& t : threads) {
293 t->join();
294 }
295 }

298void read_all_lines(const std::string& file_path, std::vector<std::string>& lines) {
299 std::ifstream file(file_path);
300 std::string str;
301 while (std::getline(file, str)) {
302 // remove trailing \r
303 if (!str.empty() && str.back() == '\r') {
304 str.pop_back();
305 }
306 lines.push_back(str);
