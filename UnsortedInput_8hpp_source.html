25template <typename T>
26class UnsortedInput final : public IndexStrategy<T> {
27public:
28 typedef typename IndexStrategy<T>::record_type record_type;
29 typedef typename record_type::value_type value_type;

31private:
32 typedef std::shared_ptr<Postlist<value_type> > shared_postlist;
33 typedef std::unordered_map<std::string, shared_postlist> postlist_map;
34 typedef std::shared_ptr<PostlistBuilder<value_type> > shared_pbuilder;
35 typedef std::unordered_map<std::string, shared_pbuilder> pbuilder_map;

37public:
38 explicit UnsortedInput(const Configuration& config)
39 : IndexStrategy<value_type>(config),
40 max_bucket_size_(to_bytes(config.max_memory_usage()) / 12) // best avg.
41 {
42 if (max_bucket_size_ == 0)
43 max_bucket_size_ = 10 * 1024 * 1024;
44 bucket_dir_ = util::tmpdir(this->config().index_directory());
45 bucket_fs_ = rehash(fs::path(), bucket_dir_, 2);
46 this->set_expected_record_count(config.expected_record_count());
47 }
48
49 virtual ~UnsortedInput() {
50 fs::remove_all(bucket_dir_);
51 }
52
53 virtual void index() {
54 // group / sort / store records
55 close(bucket_fs_);
56 postlist_map postlists;
57 StorageWriter<value_type> storage(this->config().index_directory());
58 const fs::directory_iterator end;
59 for (fs::directory_iterator it(bucket_dir_); it != end; ++it) {
60 if (!fs::is_regular_file(it->path())) {
61 continue;
62 }
63 util::log("Processing", it->path());
64 group(it->path(), postlists);
65 switch (this->config().value_sorting()) {
66 case value_sorting_type::ascending:
67 sort<std::less<value_type> >(postlists);
68 break;
69 case value_sorting_type::descending:
70 sort<std::greater<value_type> >(postlists);
71 break;
72 default:;
73 }
74 this->stats_.key_count += postlists.size();
75 store(postlists, storage);
76 fs::remove(it->path());
77 postlists.clear();
78 }
79 storage.close();
80 }

82private:
83 static void close(std::vector<FILE*>& fs) {
84 for (auto it(fs.begin()); it != fs.end(); ++it) {
85 util::fclose(*it);
86 }
87 }

89 static void group(const fs::path& bucket_path, postlist_map& postlists) {
90 assert(fs::exists(bucket_path));
91
92 record_type record;
93 pbuilder_map builders;
94 typename pbuilder_map::const_iterator it;
95 FILE* rfs(util::fopen(bucket_path, "rb"));
96 while (record.read(rfs)) {
97 it = builders.find(record.key());
98 if (it == builders.end()) {
99 shared_pbuilder pb(new PostlistBuilder<value_type>);
100 it = builders.insert(std::make_pair(record.key(), pb)).first;
101 }
102 it->second->push_back(record.value());
103 }
104 util::fclose(rfs);
105 for (auto it = builders.begin(); it != builders.end(); ++it) {
106 postlists.insert(std::make_pair(it->first, shared_postlist(it->second->build())));
107 }
108 }

110 static const std::vector<FILE*> rehash(const fs::path& src_dir, const fs::path& des_dir, uint32_t num) {
111 assert(fs::is_empty(des_dir));
112 // create destination bucket files
113 std::vector<FILE*> des_fs(util::next_prime(num));
114 for (unsigned i(0); i != des_fs.size(); ++i) {
115 des_fs[i] = util::fopen(des_dir / ("bucket." + util::to_string(i)), "wb");
116 }
117 if (fs::exists(src_dir)) {
118 record_type record;
119 const fs::directory_iterator end;
120 for (fs::directory_iterator it(src_dir); it != end; ++it) {
121 if (!fs::is_regular_file(it->path()))
122 continue;
123
124 FILE* src_fs(util::fopen(it->path(), "rb"));
125 while (record.read(src_fs)) {
126 record.write(des_fs[util::hash32(record.key()) % des_fs.size()]);
127 }
128 util::fclose(src_fs);
129 fs::remove(it->path()); // early deletion to safe disk space
130 }
131 }
132 return des_fs;
133 }

136 static void sort(postlist_map& postlists) {

142 static void store(postlist_map& postlists, StorageWriter<value_type>& storage) {
143 for (auto it(postlists.begin()); it != postlists.end(); ++it) {
144 storage.write(it->first, *it->second);
145 }

148 void insert_(const record_type& record) override {
149 if (!record.write(bucket_fs_[util::hash32(record.key()) % bucket_fs_.size()])) {
150 throw std::runtime_error(std::format("Cannot write record : {}", record));
151 }
152 // do rehashing, if the maximal bucket size exceeds
153 if (this->stats().total_size / bucket_fs_.size() > max_bucket_size_) {
154 util::log("Bucket size exceeded", this->stats().value_count);
155 if (this->config().expected_record_count() == 0) {
156 rehash_(2 * bucket_fs_.size());
157 } else {
158 util::log("Expected total number of records", this->config().expected_record_count());
159 // estimate the final number of buckets
160 // note: division by zero is not possible
161 const double avg_value_size = this->stats().total_size / static_cast<double>(this->stats().value_count);
162 const uint64_t expected_total_size = avg_value_size * this->config().expected_record_count();
163 const size_t new_bucket_count = 1 + expected_total_size / max_bucket_size_;
164 rehash_(std::max(new_bucket_count, 2 * bucket_fs_.size()));
165 }
166 }
167 }
168
169 void rehash_(size_t new_bucket_count) {
170 close(bucket_fs_);
171 fs::path new_dir(util::tmpdir(this->config().index_directory()));
172 util::log("Rehashing current inserted records", this->stats().value_count);
173 bucket_fs_ = rehash(bucket_dir_, new_dir, new_bucket_count);
174 util::log("New number of bucket files", bucket_fs_.size());
175 fs::remove_all(bucket_dir_);
176 bucket_dir_ = new_dir;
177 }
178
179 std::vector<FILE*> bucket_fs_;
180 fs::path bucket_dir_;
181 uint64_t max_bucket_size_;
182};
