78 void Open(const fs::path& directory, util::memory_type memory) {
79 if (IsOpen())
80 return;
81
82 const fs::path data_dir(directory / StorageWriter<T>::k_data_dir);
83 const fs::path table_dir(directory / StorageWriter<T>::k_table_dir);
84
85 // open header table
86 table_.reset(Map::Open(table_dir, memory));
87
88 // open data files
89 const size_t data_file_count(std::distance(fs::directory_iterator(data_dir), fs::directory_iterator()));
90 files_.reserve(data_file_count);
91 paths_.reserve(data_file_count);
92 for (unsigned i(0); i != data_file_count; ++i) {
93 const std::string num(util::to_string(i));
94 paths_.push_back(data_dir / (StorageWriter<T>::k_data_file + num));
95 files_.push_back(util::fopen(paths_.back(), "rb"));
96 }
97 }

99 bool ReadHead(const std::string& key, Head& head) {
100 Address address;
101 if (table_ && table_->Get(key, address) && address.e1() < files_.size()) {
102 FILE* file = files_[address.e1()];
103 const uint32_t offset = address.e2();
104 return ReadHead(file, offset, head, IsThreadSafe());
105 }
106 return false;
107 }
