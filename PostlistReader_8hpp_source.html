30 uint32_t value_count = std::numeric_limits<uint32_t>::max(),
31 uint32_t page_size = swap_type::default_pagesize) {
32 std::unique_ptr<Postlist<T>> postlist;
33
34 // load postlist head
35 Head head;
36 util::fread(&head, sizeof(head), 1, file);
37 index_begin = std::min(index_begin, head.value_count);
38 value_count = std::min(value_count, head.value_count - index_begin);
39 std::size_t begin_of_payload = util::ftell(file);
40 if (head.value_size == 0) // variable value size
41 {
42 begin_of_payload += head.value_count * sizeof(size_vector::value_type);
43
44 // read value sizes to skip [0, index_begin)
45 size_vector value_sizes(index_begin);
46 util::fread(value_sizes.data(), sizeof(size_vector::value_type), value_sizes.size(), file);
47 const std::size_t offset_of_index_begin =
48 std::accumulate(value_sizes.begin(), value_sizes.end(), begin_of_payload);
49
50 // read value sizes to use [index_begin, index_begin + value_count)
51 value_sizes.resize(value_count);
52 util::fread(value_sizes.data(), sizeof(size_vector::value_type), value_sizes.size(), file);
53 const std::size_t total_payload_size = std::accumulate(value_sizes.begin(), value_sizes.end(), 0);
54
55 // build new head
56 Head new_head;
57 new_head.value_count = value_count;
58 new_head.value_size = head.value_size;
59 new_head.total_size = total_payload_size;
60 if (total_payload_size > page_size) {
61 // leave payload in swap file
62 FILE* swap_file = util::fopen(path, "rb");
63 util::fseek(swap_file, offset_of_index_begin, SEEK_SET);
64 util::log(__FILE__, __LINE__, "util::ftell", util::ftell(swap_file));
65 const swap_type swap(swap_file);
66 postlist.reset(new Postlist<T>(new_head, swap, value_sizes));
67 } else {
68 // read payload into buffer
69 page_type page(total_payload_size);
70 util::fseek(file, offset_of_index_begin, SEEK_SET);
71 util::log((__FILE__), (__LINE__), "util::ftell", util::ftell(file));
72 page.buffer_.read(file);
73 postlist.reset(new Postlist<T>(new_head, page, value_sizes));
74 }
75 } else {
76 const std::size_t offset_of_index_begin = begin_of_payload + index_begin * head.value_size;
77 const std::size_t total_payload_size = value_count * head.value_size;
78 // build new head
79 Head new_head;
80 new_head.value_count = value_count;
81 new_head.value_size = head.value_size;
82 new_head.total_size = total_payload_size;
83 if (total_payload_size > page_size) {
84 // read payload to use from swap file
85 FILE* swap_file = util::fopen(path, "rb");
86 util::fseek(swap_file, offset_of_index_begin, SEEK_SET);
87 const swap_type swap(swap_file);
88 postlist.reset(new Postlist<T>(new_head, swap));
89 } else {
90 // read payload to use into buffer
91 page_type page(total_payload_size);
92 util::fseek(file, offset_of_index_begin, SEEK_SET);
93 page.buffer_.read(file);
94 postlist.reset(new Postlist<T>(new_head, page));
95 }
96 }
97 const std::size_t end_of_postlist = begin_of_payload + head.total_size;
98 util::fseek(file, end_of_postlist, SEEK_SET);
99 return postlist;
100 }
