38// -----------------------------------------------------------------------------
39// Builds netspeak index with given command line options.
40// An n-gram collection consists of a set of n-gram files, whose lines must be
41// formatted as follows (other formats are not supported):
42//
43// <n-gram> TAB <n-gram-frequency>
44//
45// The separator between the n-gram and its frequency is a single tab ('\t').
46// The separator between the n-grams words is a single whitespace (' ').
47// Example:
48//
49// hello world 123
50// -----------------------------------------------------------------------------
51int BuildCommand::run(boost::program_options::variables_map variables) {
52 const std::string org_input_dir = variables["in"].as<std::string>();
53 const std::string org_output_dir = variables["out"].as<std::string>();
54
55 fs::path input_dir(org_input_dir);
56 fs::path output_dir(org_output_dir);
57
58 if (variables.count("merge-duplicates")) {
59 // if keys have to be merged and no merge dir was given. Create 'tmp'
60 // dir in dst_folder
61 fs::path tmp_dir = variables.count("merge-dir") ? fs::path(variables["merge-dir"].as<std::string>())
62 : output_dir / fs::path("tmp");
63
64 netspeak::MergeDuplicates(input_dir, tmp_dir);
65 input_dir = tmp_dir;
66 output_dir /= "tmp-index";
67 }
68
69 netspeak::BuildNetspeak(input_dir, output_dir);
70 if (org_input_dir != input_dir.c_str()) {
71 if (input_dir.parent_path() == org_input_dir) {
72 fs::remove_all(input_dir);
73 }
74 fs::rename(output_dir, fs::path(org_output_dir));
75 }
76
77 return EXIT_SUCCESS;
78}
