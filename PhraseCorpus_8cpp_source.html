83std::vector<Phrase> PhraseCorpus::read_phrases(const std::vector<Phrase::Id>& phrase_ids) const {
84 // The gist of this method is the following:
85 // We use async IO to read all the raw phrases data in parallel into memory
86 // and then we decode the raw data sequentially. This is a lot faster than
87 // using sync IO.
88 //
89 // Instead of making many small buffers, we will go through all phrase ids
90 // once and figure out how much memory we need in total and allocate that in
91 // one block.
92
93 size_t total_mem = 0;
94 for (const auto& id : phrase_ids) {
95 total_mem += entry_size(id.length());
96 }
97
98 // allocate buffer
99 std::unique_ptr<char[]> buffer(new char[total_mem]);
100
101 const size_t count = phrase_ids.size();
102
103 // prepare objects for async IO
104 std::vector<aiocb> aio_read_objs(count);
105 std::vector<aiocb*> aio_read_ptrs(count);
106 std::memset(aio_read_objs.data(), 0, count * sizeof(aiocb));
107
108 size_t buffer_pos = 0;
109 for (size_t i = 0; i < count; ++i) {
110 const auto id = phrase_ids[i];
111 const auto size = entry_size(id.length());
112 const auto fd_it = fd_map.find(id.length());
113 util::check(fd_it != fd_map.end(), __func__);
114
115 aio_read_objs[i].aio_buf = &(buffer[buffer_pos]);
116 aio_read_objs[i].aio_nbytes = size;
117 aio_read_objs[i].aio_fildes = fd_it->second;
118 aio_read_objs[i].aio_offset = id.local() * size;
119 aio_read_objs[i].aio_lio_opcode = LIO_READ;
120 aio_read_ptrs[i] = &aio_read_objs[i];
121
122 buffer_pos += size;
123 }
124
125 util::check(::lio_listio(LIO_WAIT, aio_read_ptrs.data(), count, nullptr) != -1, __func__, "lio_listio failed");
126
127 std::vector<Phrase> phrases;
128 phrases.reserve(count);
129
130 for (size_t i = 0; i < count; ++i) {
131 const auto aio_ptr = aio_read_ptrs[i];
132 util::check(::aio_return(aio_ptr) == static_cast<ssize_t>(aio_ptr->aio_nbytes), __func__, "aio_read64 failed");
133
134 phrases.push_back(decode_((const char*)aio_ptr->aio_buf, phrase_ids[i]));
135 }
136 return phrases;
137}

141Phrase PhraseCorpus::decode_(const char* buffer, Phrase::Id id) const {
142 Phrase::Frequency freq;
143 buffer = value_traits<Phrase::Frequency>::copy_from(freq, buffer);
144
145 Phrase phrase(id, freq);
146
147 const auto end = buffer + sizeof(WordId) * (size_t)id.length();
148 WordId word_id;
149 while (buffer != end) {
150 buffer = value_traits<WordId>::copy_from(word_id, buffer);
151 std::string word(id_map->get_c_str(*(id_map->find_for_id(word_id))));
152 phrase.words().push_back(std::move(word));
153 }
154
155 return phrase;
156}

186void PhraseCorpus::open_phrase_files_(const fs::path& phrase_dir) {
187 fs::directory_iterator end;
188 Phrase::Id::Length max = 0;
189
190 for (fs::directory_iterator it(phrase_dir); it != end; ++it) {
191 const auto& path = it->path();
192 const auto filename = path.filename().string();
193 const auto phrase_len = parse_phrase_filename(filename);
194
195 if (phrase_len) {
196 fd_map.insert(std::make_pair(*phrase_len, util::FileDescriptor::open(path.string(), O_RDONLY)));
197 max = std::max(max, *phrase_len);
198 }
199 }
200
201 max_length_ = max;
202}
