60 ExternalMap(const fs::path& mph_file, const fs::path& dat_file) : Base(mph_file) {

64 inline bool ReadEntry(uint64_t offset, Entry& entry, std::false_type) const {

76 static ExternalMap* Open(const fs::path& idx_file) {
78 std::ifstream ifs(idx_file);
79 if (!ifs) {
80 throw std::runtime_error(std::format("Cannot open : {}", idx_file));
81 }
82 std::string mph_file;
83 std::string dat_file;
84 std::getline(ifs, mph_file);
85 std::getline(ifs, dat_file);
86 return new ExternalMap(idx_file.parent_path() / mph_file, idx_file.parent_path() / dat_file);
87 }

89 bool Get(const std::string& key, Value& value) {
90 Entry entry;
91 const uint64_t offset = Base::Hash(key) * EntryTraits::size_of(entry);
92
93 // Seek to table entry on disk and read it.
94 if (!ReadEntry(offset, entry, IsThreadSafeSwitch())) {
95 DEBUG_LOG("Key", key);
96 DEBUG_LOG("File offset", offset);
97 throw std::runtime_error("EntryTraits::read_from failed");
98 }
99 // Compute checksum and reject value on mismatch.
100 const Checksum checksum(util::hash<Checksum>(key));
101 if (entry.e1() != checksum) {
102 // This is no error, unknown keys normally cause a hash collision.
103 DEBUG_LOG("Key", key);
104 DEBUG_LOG("File offset", offset);
105 DEBUG_LOG("Actual checksum", checksum);
106 DEBUG_LOG("Expected checksum", entry.e1());
107 return false;
108 }
109 value = entry.e2();
110 return true;
111 }
112
113private:
114 FILE* data_;
115 mutable std::mutex mutex_;
116};
