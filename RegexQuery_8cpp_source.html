10/*
11 * The implementation will do the following optimizations:
12 *
13 * 1. `q째` == `째q` == `q`
14 * 2. `*` == `**`
15 * 3. `?*` == `*?`
16 * 4. `[aa]` == `[a]` == `a`
17 * 5. `()` == `째`
18 * 6. `(w)*` == `*(w)` == `*`
19 *
20 * Syntax:
21 * - 째 is the empty word
22 * - q is a query
23 * - [abc] is a char set
24 * - (abc) is an optional word
25 */

33uint32_t RegexQuery::combinations_upper_bound() const {
34 uint64_t count = 1;
35 for (const auto& unit : units) {
36 switch (unit.type) {
37 case RegexUnit::Type::QMARK:
38 case RegexUnit::Type::STAR:
39 return std::numeric_limits<uint32_t>::max();
40
41 case RegexUnit::Type::OPTIONAL_WORD:
42 count *= 2;
43 break;
44
45 case RegexUnit::Type::CHAR_SET:
46 count *= unit.value.size();
47 break;
48
49 default:
50 break;
51 }
52
53 if (count == 0)
54 return 0;
55 if (count >= std::numeric_limits<uint32_t>::max())
56 return std::numeric_limits<uint32_t>::max();
57 }
58 return count;
59}

75size_t RegexQuery::min_utf8_input_length() const {
76 size_t min = 0;
77 for (const auto& unit : units) {
78 switch (unit.type) {
79 case RegexUnit::Type::WORD:
80 for (const auto c : unit.value) {
81 min += number_of_utf8_bytes(c);
82 }
83 break;
84
85 case RegexUnit::Type::QMARK:
86 min += MIN_UTF8_BYTES_PER_CHAR;
87 break;
88
89 case RegexUnit::Type::CHAR_SET: {
90 if (unit.value.empty()) {
91 return std::numeric_limits<size_t>::max();
92 }
93 size_t char_set_min = std::numeric_limits<size_t>::max();
94 for (const auto c : unit.value) {
95 const auto c_min = number_of_utf8_bytes(c);
96 if (c_min < char_set_min) {
97 char_set_min = c_min;
98 if (char_set_min == MIN_UTF8_BYTES_PER_CHAR) {
99 break;
100 }
101 }
102 }
103 min += char_set_min;
104 break;
105 }
106
107 default:
108 // ignore all other tokens
109 break;
110 }
111 }
112 return min;
113}
114size_t RegexQuery::max_utf8_input_length() const {
115 size_t max = 0;
116 for (const auto& unit : units) {
117 switch (unit.type) {
118 case RegexUnit::Type::WORD:
119 case RegexUnit::Type::OPTIONAL_WORD:
120 for (const auto c : unit.value) {
121 max += number_of_utf8_bytes(c);
122 }
123 break;
124
125 case RegexUnit::Type::QMARK:
126 max += MAX_UTF8_BYTES_PER_CHAR;
127 break;
128
129 case RegexUnit::Type::CHAR_SET: {
130 // it's not defined what should happen if the char_set doesn't contain
131 // characters, so we just do whatever
132 size_t char_set_max = 0;
133 for (const auto c : unit.value) {
134 const auto c_max = number_of_utf8_bytes(c);
135 if (c_max > char_set_max) {
136 char_set_max = c_max;
137 if (char_set_max == MAX_UTF8_BYTES_PER_CHAR) {
138 break;
139 }
140 }
141 }
142 max += char_set_max;
143 break;
144 }
145
146 case RegexUnit::Type::STAR:
147 return std::numeric_limits<size_t>::max();
148 }
149 }
150 return max;

168void remove_trailing_optional_words(std::vector<RegexUnit>& units) {
169 // remove qmarks and optional words
170
171 size_t qmarks = 0;
172 while (!units.empty()) {
173 const auto type = units.back().type;
174 if (type == RegexUnit::Type::QMARK) {
175 qmarks++;
176 units.pop_back();
177 } else if (type == RegexUnit::Type::OPTIONAL_WORD) {
178 units.pop_back();
179 } else {
180 // fun's over. We hit a unit which isn't a QMARK or optional word
181 break;
182 }
183 }
184
185 // add removed qmarks back in
186 // (all qmarks are equivalent, so we'll just create new ones)
187
188 while (qmarks-- > 0) {
189 units.push_back(RegexUnit::qmark());
190 }
191}

202std::u32string without_duplicates(const std::u32string& characters) {
203 if (characters.size() < 2) {
204 return std::u32string(characters);
205 }
206
207 std::vector<char32_t> chars(characters.begin(), characters.end());
208 std::sort(chars.begin(), chars.end());
209
210 std::u32string dupFree;
211 char32_t last = chars[0];
212 dupFree.push_back(last);
213 for (size_t i = 1; i < chars.size(); i++) {
214 char32_t ch = chars[i];
215 if (ch != last) {
216 last = ch;
217 dupFree.push_back(last);
218 }
219 }
220
221 return dupFree;
222}

238void RegexQueryBuilder::add(RegexUnit unit) {
239 if (reject_all) {
240 // we can't recover from the reject_all state, so we don't need to done
241 // anything anymore
242 return;
243 }
244
245 unit = optimize_unit(unit);
246
247 if (unit.value.empty()) {
248 // there's no point in concatenating the empty string
249 if (unit.type == RegexUnit::Type::WORD || unit.type == RegexUnit::Type::OPTIONAL_WORD) {
250 return; // rule 1, 5
251 }
252
253 // make reject all
254 if (unit.type == RegexUnit::Type::CHAR_SET) {
255 reject_all = true;
256 return;
257 }
258 }
259
260 // if the vector is empty, just add the unit
261 if (units.empty()) {
262 units.push_back(unit);
263 return;
264 }
265
266 // using the above rules, we will try to eliminate and simplify as many units
267 // as possible.
268
269 // The order of `?`s and `*`s generally doesn't matter but we will strictly
270 // enforce the order `?` < `*`. This will actually make optimizing easier.
271
272 auto prev = units.back();
273 switch (unit.type) {
274 case RegexUnit::Type::WORD:
275 if (prev.type == RegexUnit::Type::WORD) {
276 // concatenate adjacent words.
277 std::u32string concat(prev.value);
278 concat.append(unit.value);
279 units.back() = RegexUnit::word(concat);
280 return;
281 }
282 break;
283
284 case RegexUnit::Type::QMARK:
285 if (prev.type == RegexUnit::Type::STAR) {
286 // switch their order (rule 3)
287 units.back() = unit;
288 units.push_back(prev);
289 return;
290 }
291 break;
292
293 case RegexUnit::Type::STAR:
294 if (prev.type == RegexUnit::Type::STAR) {
295 return; // rule 2
296 }
297 if (prev.type == RegexUnit::Type::OPTIONAL_WORD || prev.type == RegexUnit::Type::QMARK) {
298 // rule 6, 3
299 remove_trailing_optional_words(units);
300 units.push_back(unit);
301 return;
302 }
303 break;
304
305 case RegexUnit::Type::CHAR_SET:
306 // char_sets are already optimized as is, so we don't have to do anything
307 break;
308
309 case RegexUnit::Type::OPTIONAL_WORD:
310 if (prev.type == RegexUnit::Type::STAR) {
311 return; // rule 6
312 }
313 break;
314 }
315
316 // just add the unit
317 units.push_back(unit);
318}
