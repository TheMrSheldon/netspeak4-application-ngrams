183std::string create_regex_pattern(const RegexQuery& query) {
184 std::u32string pattern;
185
186 for (const auto& unit : query.get_units()) {
187 switch (unit.type) {
188 case RegexUnit::Type::QMARK:
189 // https://stackoverflow.com/a/13163802/7595472
190 pattern.append(
191 U"(?:[\\x00-\\x7F]|(?:[\\xC2-\\xDF]|\\xE0[\\xA0-\\xBF]|\\xED["
192 U"\\x80-"
193 U"\\x9F]|(?:[\\xE1-\\xEC]|[\\xEE-\\xEF]|\\xF0[\\x90-\\xBF]|["
194 U"\\xF1-"
195 U"\\xF3][\\x80-\\xBF]|\\xF4[\\x80-\\x8F])[\\x80-\\xBF])[\\x80-"
196 U"\\xBF])");
197 break;
198
199 case RegexUnit::Type::STAR:
200 pattern.append(U".*");
201 break;
202
203 case RegexUnit::Type::CHAR_SET: {
204 pattern.append(U"(?:");
205 bool first = true;
206 for (const auto c : unit.value) {
207 if (first) {
208 first = false;
209 } else {
210 pattern.push_back('|');
211 }
212 regex_pattern_append_char(pattern, c);
213 }
214 pattern.append(U")");
215 break;
216 }
217
218 case RegexUnit::Type::OPTIONAL_WORD:
219 pattern.append(U"(?:");
220 for (const auto c : unit.value) {
221 regex_pattern_append_char(pattern, c);
222 }
223 pattern.append(U")?");
224 break;
225
226 default:
227 for (const auto c : unit.value) {
228 regex_pattern_append_char(pattern, c);
229 }
230 break;
231 }
232 }
233
234 std::wstring_convert<std::codecvt_utf8<char32_t>, char32_t> conv;
235 return conv.to_bytes(pattern);

241
242 // split vocabulary into words
243 std::string::size_type pos;
244 std::string::size_type prev = 0;
245 while ((pos = vocabulary_.find('\n', prev)) != std::string::npos) {
246 auto const length = pos - prev;
247 if (length > 0) {
248 WordEntry entry = { (uint32_t)prev, (uint16_t)length };
249 words_.push_back(entry);
250 }
251 prev = pos + 1;
252 }

261void DefaultRegexIndex::initialize_all_chars() {
262 // Since C++ provides no easy way to just iterate all UTF-32 characters of a
263 // UTF-8 string, we use the std::string to std::u23string methods on
264 // substrings of the whole vocabulary. This will keep the memory overhead
265 // minimal while still being reasonably performant.
266
267 std::wstring_convert<std::codecvt_utf8<char32_t>, char32_t> conv;
268
269 // resever a few slots for good measure
270 all_chars_ = std::unordered_set<char32_t>();
271 all_chars_.reserve(10000);
272
273 // one substring will have this many words.
274 size_t group_size = 1024;
275 for (size_t i = 0; i < words_.size(); i += group_size) {
276 size_t from = (size_t)words_[i].offset;
277 size_t to;
278 if (i + group_size < words_.size()) {
279 to = words_[i + group_size].offset;
280 } else {
281 to = vocabulary_.size();
282 }
283
284 const std::string& sub = vocabulary_.substr(from, to - from);
285 std::u32string u32 = conv.from_bytes(sub);
286 all_chars_.insert(u32.begin(), u32.end());
287 }
288
289 all_chars_.erase('\n');
290 all_chars_.erase('\r');
291}

293void DefaultRegexIndex::initialize_word_hash_table() {
294 // the hash table will be implemented via linear probing, so we need enough
295 // spaces between entries.
296 size_t n = next_pow_of_2(words_.size());
297 if (n < words_.size() + (words_.size() / 2))
298 n *= 2; // n is too small, so let's make it bigger
299
300 word_hash_table_ = std::vector<uint32_t>(n);
301
302 // initialize all slots of the hash table as empty
303 for (size_t i = 0; i < n; i++) {
304 word_hash_table_[i] = UINT32_MAX;
305 }
306
307 // insert all words
308 const uint32_t mask = n - 1;
309 for (uint32_t index = 0, count = words_.size(); index < count; index++) {
310 const auto entry = words_[index];
311 uint32_t hash = hash_word(vocabulary_, entry.offset, entry.length) & mask;
312
313 while (true) {
314 if (word_hash_table_[hash] == UINT32_MAX) {
315 // empty slot
316 word_hash_table_[hash] = index;
317 break;
318 }
319
320 // linear probing
321 hash = (hash + 1) & mask;
322 }
323 }
324}

337RegexQuery DefaultRegexIndex::optimize_query(const RegexQuery& query) const {
338 RegexQueryBuilder builder;
339
340 for (const auto& unit : query.get_units()) {
341 const std::u32string& value = unit.value;
342
343 switch (unit.type) {
344 case RegexUnit::Type::WORD:
345 if (contains_unknown_characters(value, all_chars_)) {
346 // add the empty set causing the query to reject all words
347 builder.add(RegexUnit::char_set(U""));
348 } else {
349 // just add the word
350 builder.add(unit);
351 }
352 break;
353
354 case RegexUnit::Type::OPTIONAL_WORD:
355 if (contains_unknown_characters(value, all_chars_)) {
356 // An optional word which contains an unknown character cannot be
357 // matched, so we don't add it to the new query.
358 } else {
359 // just add the optional word
360 builder.add(unit);
361 }
362 break;
363
364 case RegexUnit::Type::CHAR_SET: {
365 if (contains_unknown_characters(value, all_chars_)) {
366 // remove all unknown character from the character set
367 std::unordered_set<char32_t> set;
368 for (const auto& c : value) {
369 if (all_chars_.find(c) != all_chars_.end()) {
370 set.insert(c);
371 }
372 }
373 std::u32string new_value(set.begin(), set.end());
374 builder.add(RegexUnit::char_set(new_value));
375 } else {
376 // just add the character set
377 builder.add(unit);
378 }
379 break;
380 }
381
382 default:
383 builder.add(unit);
384 break;
385 }
386 }
387
388 return builder.to_query();
389}

398uint32_t DefaultRegexIndex::find_word(const std::string& word) const {
399 const uint32_t mask = word_hash_table_.size() - 1;
400 uint32_t hash = hash_word(word, 0, word.size()) & mask;
401
402 while (true) {
403 auto index = word_hash_table_[hash];
404 if (index == UINT32_MAX)
405 return UINT32_MAX; // not found
406
407 // found a candidate
408 const auto entry = words_[index];
409
410 // compare words
411 if (entry.length == word.size() && vocabulary_.compare(entry.offset, entry.length, word) == 0) {
412 // found the given word in the vocabulary
413 return index;
414 }
415
416 // the candidate isn't the given word
417 hash = (hash + 1) & UINT32_MAX;
418 }

434 std::vector<utf8_finite_regex_unit> result;
435
436 std::wstring_convert<std::codecvt_utf8<char32_t>, char32_t> conv;

466void match_query_hash_lookup_impl(const std::vector<utf8_finite_regex_unit>& stack, size_t stack_index,
467 std::string& word, std::vector<uint32_t>& matches,
468 std::function<uint32_t(const std::string&)>& find_word) {
469 if (stack_index >= stack.size()) {
470 // we reached the end of the stack, so just check the word
471 const uint32_t index = find_word(word);
472 if (index != UINT32_MAX) {
473 // found a word
474 matches.push_back(index);
475 }
476 } else {
477 // go through all combinations
478 const auto& current = stack[stack_index];
479 for (const std::string& alternative : current.alternatives) {
480 const auto len = word.size();
481 word.append(alternative);
482 // recursively call the function
483 match_query_hash_lookup_impl(stack, stack_index + 1, word, matches, find_word);
484 word.erase(len); // remove all appended characters
485 }
486 }

488void DefaultRegexIndex::match_query_hash_lookup(const RegexQuery& query, std::vector<std::string>& matches,
489 uint32_t max_matches) const {
490 const auto stack = finite_query_to_utf8(query);
491 std::string temp_word;
492 std::vector<uint32_t> indexes;
493 std::function<uint32_t(const std::string&)> find_word_fn = [&](const std::string& word) { return find_word(word); };
494 match_query_hash_lookup_impl(stack, 0, temp_word, indexes, find_word_fn);
495
496 // sort indexes
497 std::sort(indexes.begin(), indexes.end());
498
499 const auto count = indexes.size();
500 if (count) {
501 // add words without duplicates
502 uint32_t last_index = indexes[0];
503 matches.push_back(word_at_index(last_index));
504 uint32_t added = 1;
505 for (size_t i = 1; i < count && added < max_matches; i++) {
506 const auto index = indexes[i];
507 if (index != last_index) {
508 last_index = index;
509 matches.push_back(word_at_index(last_index));
510 added++;
511 }
512 }
513 }

516void DefaultRegexIndex::match_query_regex(const RegexQuery& query, std::vector<std::string>& matches,
517 uint32_t max_matches, std::chrono::nanoseconds timeout) const {
518 // build a regex and match it against all words.
519 const boost::regex expression(create_regex_pattern(query));
520 const uint16_t min_length = query.min_utf8_input_length();
521 const uint16_t max_length = query.max_utf8_input_length();
522
523 const auto start = std::chrono::steady_clock::now();
524 uint32_t matches_added = 0;
525 for (size_t i = 0; i < words_.size(); i++) {
526 const auto entry = words_[i];
527
528 if (entry.length < min_length || entry.length > max_length) {
529 // we can reject this word based on length alone
530 continue;
531 }
532
533 const auto begin = vocabulary_.begin() + entry.offset;
534 const auto end = vocabulary_.begin() + (entry.offset + entry.length);
535 if (boost::regex_match(begin, end, expression)) {
536 // found a match
537 matches.push_back(word_from_entry(entry));
538 matches_added++;
539 if (matches_added >= max_matches)
540 break; // done
541 }
542
543 // check the time every now and then
544 if (i % 256 == 0) {
545 if ((std::chrono::steady_clock::now() - start) > timeout)
546 break; // timeout
547 }
548 }
549}

552void DefaultRegexIndex::match_query(const RegexQuery& query, std::vector<std::string>& matches, uint32_t max_matches,
553 std::chrono::nanoseconds timeout) const {
554 // no matches are required
555 if (max_matches == 0) {
556 return;
557 }
558
559 const RegexQuery& q = optimize_query(query);
560
561 if (query.accept_all_non_empty()) {
562 // the query will match all words
563 // Note: Technically, the empty word may be part of the index, but that's
564 // unlikely because it doesn't make sense in the context of Netspeak.
565 uint32_t to_add = std::min(max_matches, (uint32_t)words_.size());
566 for (uint32_t i = 0; i < to_add; i++) {
567 matches.push_back(word_at_index(i));
568 }
569 return;
570 }
571
572 if (query.reject_all()) {
573 // if the query will reject all words anyway, so there's nothing to do.
574 return;
575 }
576
577 if (q.combinations_upper_bound() < 1000) {
578 // the query can match only finitely many words
579 // We can reasonably assume that we're going to be faster than any
580 // timeout, so we will just ignore it.
581 match_query_hash_lookup(query, matches, max_matches);
582 return;
583 }
584
585 // do regex matching
586 match_query_regex(query, matches, max_matches, timeout);
587}
