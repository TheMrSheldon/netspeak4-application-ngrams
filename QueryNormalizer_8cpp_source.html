40 static void add_to_concat(SimpleQuery::Unit& concat, SimpleQuery::Unit unit) {
41 // If a concat is added to a concat, then we can just inline the elements of
42 // the child concat.
43 if (unit.tag() == SimpleQuery::Unit::Tag::CONCAT) {
44 for (auto& child : unit.drain_children()) {

55 for (auto& child : unit.drain_children()) {

99 SimpleQuery::Unit dict_to_simple(const QueryPtr& query, const UnitPtr& unit) {
100 const auto source = to_source(query, unit);
101
102 const std::string& text = get_text(unit);
103 auto text_unit = SimpleQuery::Unit::new_word(text, source);
104
105 if (!dictionary) {
106 // no dictionary for this index
107 return text_unit;
108 }
109
110 const auto range = dictionary->equal_range(text);
111 if (range.first == range.second) {
112 // the word isn't in the dictionary, hence no synonyms
113 return text_unit;
114 }
115
116 // make an alternation with the text itself and all of its synonyms
117 // (be sure to ignore the original word just in case)
118
119 auto alt = SimpleQuery::Unit::new_alternation(source);
120 alt.add_child(std::move(text_unit));
121
122 std::vector<std::string> words; // just a temporary buffer
123
124 for (auto it = range.first; it != range.second; it++) {
125 const auto& syn = it->second;
126 if (syn == text) {
127 continue;
128 }
129
130 // synonyms may be phrases, so e.g. `#fail` might map to "break down", so
131 // we have to split the synonym into words first
132 words.clear();
133 boost::split(words, syn, [](char c) { return c == ' '; });
134
135 if (words.size() == 1) {
136 // the synonym is just one word
137 alt.add_child(SimpleQuery::Unit::new_word(words[0], source));
138 } else {
139 // the synonym is a phrase
140 auto concat = SimpleQuery::Unit::new_concat(source);
141 for (const auto& word : words) {
142 concat.add_child(SimpleQuery::Unit::new_word(word, source));
143 }
144 alt.add_child(std::move(concat));
145 }
146 }
147
148 return alt;
149 }

168 SimpleQuery::Unit order_to_simple(const QueryPtr& query, const UnitPtr& unit) {
169 const auto source = to_source(query, unit);
170
171 // Order sets easily blow up simple queries because an order set with n
172 // element will be converted to (n+1)! elements (n! copies of the original n
173 // elements + n! concat elements). Some blow-up is necessary, because that's
174 // how the order set works, but the normalization options to prevent
175 // malicious queries that could crash or block our server.
176 //
177 // 1) options.max_length:
178 // We know that all generated concats have the same min_length (because
179 // min_length doesn't depend on the order of the elements in the concat), so
180 // if a concat of all elements of the order set has a min_length >
181 // options.max_length, then we know that all norm queries generated from
182 // this order set will be discarded anyway. Thus we don't need to do any
183 // work and can just return an empty alternation.
184 //
185 // 2) options.max_norm_queries:
186 // We know that we will create n! concats, so have to assume all of them
187 // will translate into norm queries, creating n! many. If n! >
188 // options.max_norm_queries, we can abort early because the input query will
189 // (very likely) create too many norm queries.
190
191 std::vector<SimpleQuery::Unit> elements;
192 uint32_t elements_min_length = 0;
193 for (const auto& child : unit->children()) {
194 const auto child_simple = unit_to_simple(query, child);
195 const auto range = child_simple.length_range();
196 if (range.empty()) {
197 // any concat with a unit that doesn't accept any phrases will also not
198 // accept any phrases
199 return SimpleQuery::Unit::new_alternation(source);
200 } else {
201 elements_min_length += range.min;
202 }
203
204 if (range.max == 0) {
205 // the child is equal to the empty concat and can therefore be removed
206 continue;
207 }
208 elements.push_back(unit_to_simple(query, child));
209 }
210
211 // check 1)
212 if (elements_min_length > options.max_length) {
213 return SimpleQuery::Unit::new_alternation(source);
214 }
215
216 // check 2)
217 if (elements.size() >= 7 || util::factorial(elements.size()) > options.max_norm_queries) {
218 // 7! will result in at least 7! (~5k) norm queries and at least 8! (~40k)
219 // elements, so it's used as a hard limit. Our current indexes only
220 // support up to 5-grams, so this shouldn't be an issue.
221
222 throw invalid_query_error("Orderset with too many options.");
223 }
224
225 // Special cases for 0 and 1 many elements.
226
227 if (elements.size() == 0) {
228 // A empty order set is equal to the empty concat.
229 return SimpleQuery::Unit::new_concat(source);
230 }
231 if (elements.size() == 1) {
232 // An order set with one unit is equal to its unit.
233 return std::move(elements[0]);
234 }
235
236 // This is the general case.
237 // Make an alternative that contains the all permutations.
238
239 auto alt = SimpleQuery::Unit::new_alternation(source);
240
241 std::vector<size_t> indexes;
242 for (size_t i = 0; i < elements.size(); i++) {
243 indexes.push_back(i);
244 }
245
246 // go through all permutations of indexes
247
248 // All the permutating is done by std::next_permutation.
249 // https://en.cppreference.com/w/cpp/algorithm/next_permutation
250 do {
251 auto perm_concat = SimpleQuery::Unit::new_concat(source);
252
253 for (const auto i : indexes) {
254 add_to_concat(perm_concat, elements[i].clone());
255 }
256
257 alt.add_child(std::move(perm_concat));
258 } while (std::next_permutation(indexes.begin(), indexes.end()));
259
260 return alt;
261 }

370 static bool contains_child(const SimpleQuery::Unit& unit, const SimpleQuery::Unit::Tag tag) {

372 [&tag](const SimpleQuery::Unit& u) { return u.tag() == tag; });

375 static void inline_nested(SimpleQuery::Unit& unit, const SimpleQuery::Unit::Tag tag) {
376 // check if there are nested units to inline
377 if (contains_child(unit, tag)) {
378 for (auto& child : unit.drain_children()) {
379 if (child.tag() == tag) {
380 for (auto& nested_child : child.drain_children()) {
381 unit.add_child(std::move(nested_child));
382 }
383 } else {
384 unit.add_child(std::move(child));
385 }

390 static void concat_star_optimization(SimpleQuery::Unit& concat) {
391 // Star optimization
392 //
393 // The basic idea here is that the star operator has the power to "absorb"
394 // other units. I.e. the star in `the * [ foo ]` can absorb the optional
395 // word "foo" meaning that this query is equivalent to `the *`.
396 //
397 // More generally, the star can absorb any adjacent unit with a minimum
398 // length of 0 (e.g. optional word, other stars). The limiting factor then
399 // becomes the "adjacent" condition but we can switch the order of stars and
400 // QMarks, so we can potentially absorb even more units. E.g. `* ? [ foo ]`
401 // == `? * [ foo ]` == `? *` == `* ?`.
402 //
403 // Note: The relative order of child units is not actually changed. This is
404 // important because the source backtracking can become very confusing
405 // otherwise (with units unexpectedly changing positions).
406
407 // The whole thing is implemented in 2 passes, so we cache the min length
408 std::vector<uint32_t> min_lengths;
409 for (const auto& child : concat.children()) {
410 min_lengths.push_back(child.min_length());
411 }
412
413 // A list of the indexes of all children to be removed
414 std::vector<size_t> to_remove;
415
416 auto mark_for_removal = [&min_lengths, &to_remove, &concat](size_t star_index, bool forward) {
417 size_t processed = 0;
418 size_t index = star_index;
419
420 auto move_next = [&]() {
421 // change the index and check the bounds of the vector
422 if (forward) {
423 index++;
424 return index < concat.children().size();
425 } else {
426 if (index == 0) {
427 return false;
428 } else {
429 index--;
430 return true;
431 }
432 }
433 };
434
435 while (move_next()) {
436 processed++;
437 const auto& child = concat.children()[index];
438 if (child.tag() == SimpleQuery::Unit::Tag::QMARK) {
439 // skip qmarks.
440 } else if (min_lengths[index] == 0) {
441 // we found a unit to absorb
442 to_remove.push_back(index);
443 } else {
444 // we reached a unit we can neither absorb nor skip (e.g. a word),
445 // so we'll stop here and search for another star
446 break;
447 }
448 }
449
450 return processed;
451 };
452 auto remove = [&min_lengths, &to_remove, &concat]() {
453 // sort indexes in descending order
454 std::sort(to_remove.begin(), to_remove.end(), std::greater<size_t>());
455
456 for (auto index : to_remove) {
457 min_lengths.erase(min_lengths.begin() + index);
458 concat.remove_child(index);
459 }
460
461 to_remove.clear();
462 };
463
464 // First pass: Remove all zero-min-length units preceeded by a star
465 for (size_t i = 0; i < concat.children().size(); i++) {
466 const auto& star = concat.children()[i];
467 if (star.tag() == SimpleQuery::Unit::Tag::STAR) {
468 i += mark_for_removal(i, true);
469 }
470 }
471 remove();
472
473 // Second pass: This is basically the same as the first pass, but we will
474 // remove all zero-min-length units followed by a star
475 for (size_t i = concat.children().size(); i > 0; i--) {
476 const auto& star = concat.children()[i - 1];
477 if (star.tag() == SimpleQuery::Unit::Tag::STAR) {
478 i -= mark_for_removal(i - 1, false);
479 }
480 }
481 remove();
482 }
483 void optimize_simple_concat(SimpleQuery::Unit& unit) const {
484 optimize_simple_children(unit);
485
486 if (unit.children().empty()) {
487 // empty concats are removed by the parent
488 return;
489 }
490
491 if (unit.children().size() == 1) {
492 // replace the one-children concat with its child
493 unit = unit.pop_back();
494 return;
495 }
496
497 inline_nested(unit, SimpleQuery::Unit::Tag::CONCAT);
498
499 // check for the empty alternation
500 // The empty alternation is the absorbing element of concatenation
501 const auto empty = std::find_if(unit.children().begin(), unit.children().end(), [](const SimpleQuery::Unit& u) {
502 return u.tag() == SimpleQuery::Unit::Tag::ALTERNATION && u.children().empty();
503 });
504 if (empty != unit.children().end()) {
505 const auto empty_source = empty->source();
506 unit = SimpleQuery::Unit::new_alternation(empty_source);
507 return;
508 }
509
510 if (contains_child(unit, SimpleQuery::Unit::Tag::STAR)) {
511 concat_star_optimization(unit);
512 }
513 }
514
515 void optimize_simple_alt(SimpleQuery::Unit& unit) const {
516 optimize_simple_children(unit);
517
518 if (unit.children().empty()) {
519 // empty alt are removed by the parent
520 return;
521 }
522
523 if (unit.children().size() == 1) {
524 // replace the one-children alt with its child
525 unit = unit.pop_back();
526 return;
527 }
528
529 inline_nested(unit, SimpleQuery::Unit::Tag::ALTERNATION);
530
531 // Right now, the Netspeak Query syntax is simple enough that further
532 // optimization is unnecessary. Should we extend the query syntax or allow
533 // users to send us query ASTs directly, this should be updated to include
534 // more optimizations.
535 //
536 // One such optimization is based of min and max lengths of paths. We can
537 // eliminate certain alternatives
538 }

542 optimize_simple_children(query.root());
543 }

567 std::function<uint64_t(size_t)> estimate(const SimpleQuery& query) {
568 MinLengthMargin margin(0, 0);
569 const auto result = estimate_impl(query.root(), margin);
570 if (result.is_constant()) {
571 // the estimate is a constant
572 const auto constant = result.constant();
573 return [=](size_t) { return constant; };
574 } else {
575 // the estimate depends on the number of regex matches
576 const auto function = result.function();
577 return [=](size_t regex_matches) { return function(regex_matches); };
578 }

644 return UINT64_MAX;

646 return UINT64_MAX;

678 UnitResult estimate_impl(const SimpleQuery::Unit& unit, const MinLengthMargin margin) {
679 switch (unit.tag()) {
680 case SimpleQuery::Unit::Tag::WORD:
681 case SimpleQuery::Unit::Tag::QMARK:
682 return 1;
683
684 case SimpleQuery::Unit::Tag::STAR:
685 return 1UL + std::max<uint32_t>(0, options.max_length - margin.total());
686
687 case SimpleQuery::Unit::Tag::REGEX:
688 return UnitResult([](size_t input) { return (uint64_t)input; });
689
690 case SimpleQuery::Unit::Tag::ALTERNATION: {
691 UnitResult sum = 0;
692 for (const auto& child : unit.children()) {
693 sum = sum + estimate_impl(child, margin);
694 }
695 return sum;
696 }
697
698 case SimpleQuery::Unit::Tag::CONCAT: {
699 uint32_t uint_min_length = 0;
700 std::vector<uint32_t> min_lengths;
701 for (const auto& child : unit.children()) {
702 auto min_len = child.min_length();
703 if (min_len == UINT32_MAX) {
704 return 0;
705 }
706 uint_min_length += min_len;
707 min_lengths.push_back(min_len);
708 }
709
710 UnitResult total = 1;
711 uint32_t min_before = 0;
712 for (size_t i = 0; i != unit.children().size(); i++) {
713 const auto& child = unit.children()[i];
714 const uint32_t child_min = min_lengths[i];
715 const uint32_t min_after = uint_min_length - min_before - child_min;
716 const auto c_num = estimate_impl(child, margin + MinLengthMargin(min_before, min_after));
717 min_before += child_min;
718
719 if (c_num.is_constant() && c_num.constant() == UINT64_MAX) {
720 return UINT64_MAX;
721 }
722 if (c_num.is_constant() && c_num.constant() == 0) {
723 return 0;
724 }
725 total = total * c_num;
726 }
727 return total;
728 }
729
730 default:
731 throw tracable_logic_error("Unknown tag");
732 }
733 }

807public:

819private:
820 void concat(std::vector<StarQuery>& queries, const SimpleQuery::Unit& unit) {
821 switch (unit.tag()) {
822 case SimpleQuery::Unit::Tag::WORD: {
823 const auto u = StarQuery::Unit::new_word(std::make_shared<const std::string>(unit.text()), unit.source());
824 for (auto& q : queries) {
825 q.push_back(u);
826 }
827 break;
828 }
829 case SimpleQuery::Unit::Tag::QMARK: {
830 const auto u = StarQuery::Unit::new_qmark(unit.source());
831 for (auto& q : queries) {
832 q.push_back(u);
833 }
834 break;
835 }
836 case SimpleQuery::Unit::Tag::STAR: {
837 const auto u = StarQuery::Unit::new_star(unit.source());
838 for (auto& q : queries) {
839 q.push_back(u);
840 }
841 break;
842 }
843 case SimpleQuery::Unit::Tag::REGEX: {
844 throw tracable_logic_error("Regexes should have already been handled.");
845 }
846 case SimpleQuery::Unit::Tag::ALTERNATION: {
847 std::vector<StarQuery> alt;
848 alternation(alt, unit);
849
850 if (alt.empty()) {
851 queries.clear();
852 } else if (alt.size() == 1) {
853 const auto& alt_q = alt[0];
854 for (auto& q : queries) {
855 q.append(alt_q);
856 }
857 } else {
858 const std::vector<StarQuery> copy(queries);
859
860 queries.clear();
861 queries.reserve(copy.size() * alt.size());
862 for (const auto& before : copy) {
863 for (const auto& after : alt) {
864 if (before.min_length() + after.min_length() > options.max_length) {
865 // This is where a combinatory explosion takes place, so we try
866 // to optimize this case as much as possible by discarding star
867 // queries that are too long.
868 continue;
869 }
870 StarQuery q;
871 q.append(before);
872 q.append(after);
873 queries.push_back(std::move(q));
874 }
875 }
876 }
877 break;
878 }
879 case SimpleQuery::Unit::Tag::CONCAT: {
880 for (const auto& child : unit.children()) {
881 concat(queries, child);
882 }
883 break;
884 }
885 default:
886 throw tracable_logic_error("Unknown Tag");
887 }
888 }
889 void alternation(std::vector<StarQuery>& queries, const SimpleQuery::Unit& unit) {
890 switch (unit.tag()) {
891 case SimpleQuery::Unit::Tag::WORD: {
892 StarQuery q;
893 q.push_back(StarQuery::Unit::new_word(std::make_shared<const std::string>(unit.text()), unit.source()));
894 queries.push_back(std::move(q));
895 break;
896 }
897 case SimpleQuery::Unit::Tag::QMARK: {
898 StarQuery q;
899 q.push_back(StarQuery::Unit::new_qmark(unit.source()));
900 queries.push_back(std::move(q));
901 break;
902 }
903 case SimpleQuery::Unit::Tag::STAR: {
904 StarQuery q;
905 q.push_back(StarQuery::Unit::new_star(unit.source()));
906 queries.push_back(std::move(q));
907 break;
908 }
909 case SimpleQuery::Unit::Tag::REGEX: {
910 throw tracable_logic_error("Regexes should have already been handled.");
911 }
912 case SimpleQuery::Unit::Tag::ALTERNATION: {
913 for (const auto& child : unit.children()) {
914 alternation(queries, child);
915 }
916 break;
917 }
918 case SimpleQuery::Unit::Tag::CONCAT: {
919 std::vector<StarQuery> con;
920 con.push_back(StarQuery());
921 concat(con, unit);
922 for (const auto& q : con) {
923 if (q.min_length() <= options.max_length) {
924 // we go through all queries anyway, so we might as well check the
925 // min length
926 queries.push_back(q);
927 }
928 }
929 break;
930 }
931 default:
932 throw tracable_logic_error("Unknown Tag");
933 }

937class SimpleQueryNormalizer {
938public:
939 QueryNormalizer::Options options;
940
941 SimpleQueryNormalizer() = delete;
942 explicit SimpleQueryNormalizer(const QueryNormalizer::Options& options) : options(options) {}

956 void to_norm_queries(std::vector<NormQuery>& norm_queries, const SimpleQuery& query) {
957 StarQueryConverter star_conv(options);
958 std::vector<StarQuery> star_queries = star_conv.to_star_queries(query);
959
960 for (const auto& q : star_queries) {
961 const auto min = q.min_length();
962 const auto max = q.max_length();
963 // this means that the star query does not contain stars, so it's a valid
964 // norm query
965 if (min <= options.max_length && min >= options.min_length) {
966 if (min == max) {
967 add_simple(norm_queries, q);
968 } else {
969 add_with_stars(norm_queries, q);
970 }
971 }
972
973 if (norm_queries.size() > options.max_norm_queries) {
974 throw invalid_query_error("Too many norm queries");
975 }
976 }
977 }
978
979private:
980 void add_simple(std::vector<NormQuery>& norm_queries, const StarQuery& query) {
981 NormQuery nq;
982 for (const auto& u : query.units()) {
983 switch (u.tag) {
984 case StarQuery::Unit::Tag::WORD:
985 nq.units().push_back(NormQuery::Unit::word(u.text, u.source));
986 break;
987 case StarQuery::Unit::Tag::QMARK:
988 nq.units().push_back(NormQuery::Unit::qmark(u.source));
989 break;
990 default:
991 throw tracable_logic_error("Star is not valid here.");
992 }
993 }
994 norm_queries.push_back(nq);
995 }
996 void add_with_stars(std::vector<NormQuery>& norm_queries, const StarQuery& query) {
997 // the number of stars in the query
998 const size_t star_count = std::count_if(query.units().begin(), query.units().end(), [](const StarQuery::Unit& u) {
999 return u.tag == StarQuery::Unit::Tag::STAR;
1000 });
1001 // the available space to expand stars.
1002 const size_t available = options.max_length - query.min_length();
1003
1004 util::ChainCutter cutter(star_count, available);
1005 // The chain cutter will go through all possible cuts on a chain of a
1006 // certain length. We choose the length of the chain to be our available
1007 // space and the number of cuts to the number of stars. This means that the
1008 // first available-many segments of each iteration will tell us with how
1009 // many qMarks we have to replace each star with.
1010
1011 do {
1012 NormQuery nq;
1013 size_t star_i = 0;
1014 for (const auto& u : query.units()) {
1015 switch (u.tag) {
1016 case StarQuery::Unit::Tag::WORD:
1017 nq.units().push_back(NormQuery::Unit::word(u.text, u.source));
1018 break;
1019 case StarQuery::Unit::Tag::QMARK:
1020 nq.units().push_back(NormQuery::Unit::qmark(u.source));
1021 break;
1022 case StarQuery::Unit::Tag::STAR: {
1023 size_t qmark_count = cutter.segments()[star_i++];
1024 for (; qmark_count != 0; qmark_count--) {
1025 nq.units().push_back(NormQuery::Unit::qmark(u.source));
1026 }
1027 break;
1028 }
1029 default:
1030 throw tracable_logic_error("Unknown tag");
1031 }
1032 }
1033 if (nq.size() >= options.min_length) {
1034 // the query might be too short, so we have to check
1035 norm_queries.push_back(nq);
1036 }
1037 } while (cutter.next());
1038 }

1070size_t binary_search(size_t min, size_t max, const std::function<int(size_t)>& comparer) {
1071 while (min < max) {
1072 size_t m = min + ((max - min) >> 1) + 1;
1073 int cmp = comparer(m);
1074 if (cmp == 0) {
1075 return m;
1076 } else if (cmp < 0) {
1077 min = m;
1078 } else {
1079 max = m - 1;
1080 }
1081 }
1082 return min;
1083}

1085void normalize_regexes(SimpleQuery& query, const QueryNormalizer::Options& options,
1086 const std::shared_ptr<const regex::RegexIndex>& regex_index) {
1087 const auto regexes = find_regexes(query);
1088 if (!regexes.empty()) {
1089 // try to find out with how many words we can replace each regex and still
1090 // fall within our budget.
1091 // It's like Blackjack - try to go as high as possible but don't go over.
1092 auto estimator = NormQueryCounter(options).estimate(query);
1093
1094 // This comparer will return <= 0 if the number of regexes matches fall
1095 // which our budget and > 0 other wise.
1096 auto comparer = [&](size_t m) {
1097 auto estimate = estimator(m);
1098 if (estimate == options.max_norm_queries) {
1099 return 0;
1100 } else {
1101 return estimate < options.max_norm_queries ? -1 : 1;
1102 }
1103 };
1104
1105 // the maximum number of regex words such that we still fall <= our budget
1106 size_t best_max_matches;
1107 if (comparer(options.max_regex_matches) <= 0) {
1108 // it's quite likely that this will be the case
1109 best_max_matches = options.max_regex_matches;
1110 } else {
1111 best_max_matches = binary_search(0, options.max_regex_matches, comparer);
1112 }
1113
1114 if (best_max_matches == 0) {
1115 // By our estimate, we can't even replace the regexes with a single word
1116 // without going over the budget.
1117
1118 // TODO: What now?
1119 throw invalid_query_error("Query too complex");
1120 } else {
1121 std::vector<std::string> matches;
1122 for (auto regex_unit : regexes) {
1123 const auto regex_query = regex::parse_netspeak_regex_query(regex_unit->text());
1124 regex_index->match_query(regex_query, matches, best_max_matches, options.max_regex_time);
1125
1126 const auto source = regex_unit->source();
1127 auto alt = SimpleQuery::Unit::new_alternation(source);
1128 for (const auto& word : matches) {
1129 alt.add_child(SimpleQuery::Unit::new_word(word, source));
1130 }
1131 matches.clear();
1132
1133 *regex_unit = std::move(alt);
1134 }
1135 }
1136 }

1139void QueryNormalizer::normalize(std::shared_ptr<const Query> query, const Options& options,
1140 std::vector<NormQuery>& norm_queries) {
1141 const auto query_length_range = query->length_range();
1142 if (query_length_range.empty() || query_length_range.max < options.min_length ||
1143 query_length_range.min > options.max_length) {
1144 // In this case, we just cannot create any norm queries that fulfill the
1145 // normalization options.
1146 return;
1147 }
1148
1149 bool allow_regex =
1150 !!regex_index_ && options.max_regex_matches > 0 && options.max_regex_time > std::chrono::nanoseconds(0);
1151
1152 // convert the given query into a simpler form
1153 QuerySimplifier simplifier(options);
1154 simplifier.dictionary = dictionary_;
1155 simplifier.allow_regex = allow_regex;
1156 simplifier.lower_case = lower_case;
1157 SimpleQuery simple = simplifier.to_simple(query);
1158
1159 // optimize the simple form
1160 SimpleQueryOptimizer optimizer;
1161 optimizer.optimize(simple);
1162
1163 // replaces all regexes with a list of words
1164 normalize_regexes(simple, options, regex_index_);
1165
1166 // create norm queries
1167 SimpleQueryNormalizer normalizer(options);
1168 normalizer.to_norm_queries(norm_queries, simple);
1169}
