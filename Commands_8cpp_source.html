15void print_line(std::ostream& out, std::string line, bool skip_first_indentation, size_t indentation,
16 size_t max_width) {
17 std::string ident(indentation, ' ');
18 size_t available_space = max_width - indentation;
19
20 if (line.empty()) {
21 out << '\n';
22 return;
23 }
24
25 while (!line.empty()) {
26 if (skip_first_indentation) {
27 skip_first_indentation = false;
28 } else {
29 out << ident;
30 }
31
32 if (line.size() <= available_space) {
33 // line fits
34 out << line;
35 line.clear();
36 } else {
37 // we have to split the line
38 size_t space_pos = line.find(' ');
39 if (space_pos == std::string::npos || space_pos > available_space) {
40 // split the word.
41 out << line.substr(0, available_space);
42 line = line.substr(available_space);
43 } else {
44 // try to fit in as many words as possible
45 while (true) {
46 size_t next = line.find(' ', space_pos + 1);
47 if (next == std::string::npos || next > available_space) {
48 break;
49 } else {
50 space_pos = next;
51 }
52 }
53 // split
54 out << line.substr(0, space_pos);
55 line = line.substr(space_pos + 1);
56 }
57 }
58
59 out << '\n';
60 }

219int Commands::run(int argc, char* argv[]) {
220 if (argc < 2) {
221 print_error(commands_, "Not enough arguments.");
222 return EXIT_FAILURE;
223 }
224
225 try {
226 bpo::options_description global("");
227 auto global_init = global.add_options();
228 global_init("command", bpo::value<std::string>(), "command to execute");
229 global_init("help,h", "Print this help message.");
230 global_init("subargs", bpo::value<std::vector<std::string>>(), "Arguments for command");
231
232 bpo::positional_options_description pos_opt_desc;
233 pos_opt_desc.add("command", 1).add("subargs", -1);
234
235 auto parsed_options =
236 bpo::command_line_parser(argc, argv).options(global).positional(pos_opt_desc).allow_unregistered().run();
237
238 bpo::variables_map variables;
239 bpo::store(parsed_options, variables);
240
241 if (variables.count("command") == 0) {
242 if (variables.count("help") != 0) {
243 print_commands_help(commands_, true);
244 return EXIT_SUCCESS;
245 } else {
246 throw std::runtime_error("Missing command.");
247 }
248 }
249 std::string command_name = variables["command"].as<std::string>();
250
251 for (const auto& command_ptr : commands_) {
252 auto& command = *command_ptr;
253 if (command.name() == command_name) {
254 // might print help
255 if (variables.count("help") != 0) {
256 print_command_help(command, false);
257 return EXIT_SUCCESS;
258 }
259
260 // collect all arguments
261 std::vector<std::string> opts = bpo::collect_unrecognized(parsed_options.options, bpo::include_positional);
262 if (!opts.empty()) {
263 opts.erase(opts.begin());
264 }
265
266 try {
267 return run_command(command, opts);
268 } catch (std::exception& e) {
269 std::cerr << "Error: " << e.what() << "\n";
270 return EXIT_FAILURE;
271 }
272 }
273 }
274
275 print_error(commands_, "Unknown command \"" + command_name + "\".");
276 return EXIT_FAILURE;
277 } catch (std::exception& e) {
278 print_error(commands_, e.what());
279 return EXIT_FAILURE;
280 }
281}
