64 const auto corpora_it = corpora_map.find(key);
65 if (corpora_it == corpora_map.end()) {
66 corpora_map.emplace(key, std::pair<service::Corpus, std::string>(corpus, address));
67 } else {
68 const auto& other_corpus = corpora_it->second.first;

74 throw std::logic_error(what.str());
