68std::optional<Configuration> parse_overrides(const std::vector<std::string>& overrides) {
69 util::Config config;
70
71 for (auto& line : overrides) {
72 auto separator_index = std::min(line.find(':'), line.find('='));
73 if (separator_index == std::string::npos) {
74 throw std::logic_error(
75 "The given override is not valid. Expected a key-value pair but did "
76 "not find a key-value separator (either `:` or `=`).\n"
77 "Invalid override:\n" +
78 line);
79 }
80
81 auto key = line.substr(0, separator_index);
82 auto value = line.substr(separator_index + 1);
83 boost::trim(key);
84 boost::trim(value);
85
86 if (key == Configuration::EXTENDS) {
87 throw std::logic_error("The key `" + Configuration::EXTENDS +
88 "` connot be overridden.\n"
89 "Invalid override:\n" +
90 line);
91 }
92
93 config[key] = value;
94 }
95
96 if (config.empty()) {
97 return std::nullopt;
98 } else {
99 return Configuration(config);
100 }

130std::unique_ptr<service::NetspeakService::Service> build_service(boost::program_options::variables_map& variables) {
131 const auto& patterns = variables[CONFIG_KEY].as<std::vector<std::string>>();
132 const auto overrides = variables.count(OVERRIDE_KEY) == 0
133 ? std::nullopt
134 : parse_overrides(variables[OVERRIDE_KEY].as<std::vector<std::string>>());
135
136 auto entries = std::make_unique<std::vector<service::UniqueMap::entry>>();
137 for (const auto& pattern : patterns) {
138 std::vector<std::string> paths;
139 util::glob(pattern, paths);
140 for (const auto& path : paths) {
141 entries->push_back(load_map_entry(load_config(path, overrides)));
142 }
143 }
144
145 return std::make_unique<service::UniqueMap>(std::move(entries));
146}
