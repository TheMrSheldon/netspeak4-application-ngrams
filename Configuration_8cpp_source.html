57void Configuration::load_extends() {
58 if (config_.contains(EXTENDS)) {
59 auto extends_path = fs::path(config_.get(EXTENDS));
60 if (extends_path.empty()) {
61 return;
62 }
63 if (!base_dir_.empty()) {
64 extends_path = fs::absolute(base_dir_ / extends_path);
65 }
66 extends_ = std::make_unique<Configuration>(extends_path);
67 }
68}
69void Configuration::desugar_home() {
70 if (config_.contains(PATH_TO_HOME)) {
71 auto home_path = fs::path(config_.get(PATH_TO_HOME));
72 if (home_path.empty()) {
73 return;
74 }
75 if (!base_dir_.empty()) {
76 home_path = fs::absolute(base_dir_ / home_path);
77 }
78
79 auto set_default = [&](const std::string& key, const std::string& default_value) {
80 if (!config_.contains(key)) {
81 config_[key] = fs::weakly_canonical(fs::absolute(home_path / default_value)).string();
82 }
83 };
84
85 set_default(PATH_TO_PHRASE_CORPUS, DEFAULT_PHRASE_CORPUS_DIR_NAME);
86 set_default(PATH_TO_PHRASE_DICTIONARY, DEFAULT_PHRASE_DICTIONARY_DIR_NAME);

159bool parse_bool(const std::string& key, const std::string& value) {

202util::Config Configuration::full_config() const {
203 util::Config config;
204
205 for (auto& key : keys()) {
206 if (PATHS.find(key) == PATHS.end()) {
207 // not a path
