82 // Clears the given string and appends characters read from the given stream.
83 // The appending stops when either '\r', '\n', EOF or tuple_element_separator
84 // occur. Except of tuple_element_separator the terminating characters will
85 // not be extracted from the stream so that the outer parser is still able to
86 // detect line endings.
87 static inline void parse_from(value_type& value, std::istream& is) {
88 value.clear();
89 while (is.good()) {
90 switch (is.peek()) {
91 case '\r':
92 case '\n':
93 case std::istream::traits_type::eof():
94 return;
95 case tuple_element_separator:
96 is.get();
97 return;
98 default:
99 value.push_back(is.get());
100 }
101 }
102 }
