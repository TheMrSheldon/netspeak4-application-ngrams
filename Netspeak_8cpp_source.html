143service::Phrase::Word::Tag to_tag(const NormQuery::Unit& unit) {
144 std::shared_ptr<const Query::Unit> source = unit.source().unit;
145 while (source) {
146 switch (source->tag()) {
147 case Query::Unit::Tag::QMARK:
148 return service::Phrase::Word::WORD_FOR_QMARK;
149 case Query::Unit::Tag::STAR:
150 return service::Phrase::Word::WORD_FOR_STAR;
151 case Query::Unit::Tag::PLUS:
152 return service::Phrase::Word::WORD_FOR_PLUS;
153 case Query::Unit::Tag::REGEX:
154 return service::Phrase::Word::WORD_FOR_REGEX;
155 case Query::Unit::Tag::DICTSET:
156 return service::Phrase::Word::WORD_IN_DICTSET;
157 case Query::Unit::Tag::OPTIONSET:
158 return service::Phrase::Word::WORD_IN_OPTIONSET;
159 case Query::Unit::Tag::ORDERSET:
160 return service::Phrase::Word::WORD_IN_ORDERSET;
161 default:
162 break;
163 }
164 source = source->parent();
165 }
166 return service::Phrase::Word::WORD;

184void Netspeak::search(const service::SearchRequest& request, service::SearchResponse& response) throw() {
185 try {
186 // parse the query
187 const auto query = antlr4::parse_query(request.query());
188
189 // destruct the given request into options we can use
190 const auto option_pair = to_options(request);
191 const auto normalizer_options = option_pair.first;
192 const auto search_options = option_pair.second;
193
194 // perform the raw seach (returns phrases and phrase references)
195 auto raw_result = search_raw_(normalizer_options, search_options, query);
196 // resolve the phrase references and merge with the other phrases
197 auto phrase_result = merge_raw_result_(search_options, *raw_result);
198
199 // construct the result
200 auto response_result = response.mutable_result();
201 // add unknown words
202 for (const auto& unknown : phrase_result->unknown_words()) {
203 if (!phrase_corpus_.contains(unknown)) {
204 response_result->add_unknown_words(unknown);
205 }
206 }
207 // add phrases
208 for (const auto& phrase : phrase_result->phrases()) {
209 auto resp_phrase = response_result->add_phrases();
210 set_response_phrase(*resp_phrase, phrase);
211 }
212 } catch (const invalid_query_error& e) {
213 auto resp_error = response.mutable_error();
214 resp_error->set_kind(service::SearchResponse::Error::INVALID_QUERY);
215 resp_error->set_message(e.what());
216 } catch (const std::logic_error& e) {
217 auto resp_error = response.mutable_error();
218 resp_error->set_kind(service::SearchResponse::Error::INTERNAL_ERROR);
219 resp_error->set_message(e.what());
220 } catch (const std::exception& e) {
221 auto resp_error = response.mutable_error();
222 resp_error->set_kind(service::SearchResponse::Error::UNKNOWN);
223 resp_error->set_message(e.what());
224 } catch (...) {
225 auto resp_error = response.mutable_error();
226 resp_error->set_kind(service::SearchResponse::Error::UNKNOWN);
227 resp_error->set_message("Unknown error");
228 }
229}

232std::pair<QueryNormalizer::Options, SearchOptions> Netspeak::to_options(const service::SearchRequest& request) {
233 const auto& constraints = request.phrase_constraints();
234
235 // The default min length is 1 because our indexes don't contain the empty
236 // phrase
237 uint32_t min_len = std::max(1U, constraints.words_min());
238
239 // the maximum length of a phrase is given by the phrase corpus but may be
240 // lowered by the search request.
241 uint32_t max_len = phrase_corpus_.max_length();
242 if (constraints.words_max() != 0 && constraints.words_max() < max_len) {
243 max_len = constraints.words_max();
244 }
245
246 uint64_t max_freq = std::numeric_limits<uint64_t>::max();
247 if (constraints.frequency_max() != 0) {
248 max_freq = constraints.frequency_max();
249 }
250
251 SearchOptions s_options = {
252 .max_phrase_count = request.max_phrases(),
253 .max_phrase_frequency = max_freq,
254 .phrase_length_min = min_len,
255 .phrase_length_max = max_len,
256 // TODO: Replace these with configurable values
257 .pruning_high = 160000,
258 .pruning_low = 130000,
259 };
260
261 QueryNormalizer::Options n_options = {
262 .max_norm_queries = search_config_.max_norm_queries,
263
264 .min_length = min_len,
265 .max_length = max_len,
266
267 .max_regex_matches = search_config_.regex_max_matches,
268 .max_regex_time = search_config_.regex_max_time,
269 };
270
271 return std::make_pair(n_options, s_options);
272}

275struct ref_ {
276 std::shared_ptr<const NormQuery> query;
277 Phrase::Id id;
278 RawRefResult::Ref::Frequency freq;
279
280 ref_() = delete;
281 ref_(std::shared_ptr<const NormQuery> query, Phrase::Id id, RawRefResult::Ref::Frequency freq)
282 : query(query), id(id), freq(freq) {}
283
284 inline bool operator==(const ref_& rhs) const {
285 return id == rhs.id;
286 }
287 inline bool operator!=(const ref_& rhs) const {
288 return !(*this == rhs);
289 }
290 bool operator<(const ref_& rhs) const {
291 if (freq != rhs.freq) {
292 // sort by descending frequency
293 return freq > rhs.freq;
294 } else {
295 return id < rhs.id;
296 }
297 }

300std::unique_ptr<std::set<ref_>> merge_phrase_refs(const std::vector<RawResult::RefItem>& raw_references) {
301 auto set = std::make_unique<std::set<ref_>>();
302
303 for (const auto& ref_item : raw_references) {
304 const auto& query = ref_item.query;

306 for (const auto& raw_ref : ref_item.result->refs()) {

313std::unique_ptr<SearchResult> Netspeak::merge_raw_result_(const SearchOptions& options, const RawResult& raw_result) {
314 auto search_result = std::make_unique<SearchResult>();
315 util::vec_append(search_result->unknown_words(), raw_result.unknown_words());
316
317 size_t max_phrase_count = options.max_phrase_count;
318 if (max_phrase_count == 0) {
319 return search_result;
320 }
321
322 // Extract the top k unique phrase refs.
323 // TODO: This can done more efficiently
324 const auto unique_refs = merge_phrase_refs(raw_result.refs());
325 size_t top_k = std::min(unique_refs->size(), max_phrase_count);
326 std::vector<ref_> top_k_refs;
327 std::copy_n(unique_refs->begin(), top_k, std::back_inserter(top_k_refs));
328
329 // Read all top k phrases found by wildcard queries.
330 std::vector<Phrase::Id> tok_k_ref_ids;
331 for (const auto& ref : top_k_refs) {
332 tok_k_ref_ids.push_back(ref.id);
333 }
334 auto ref_phrases = phrase_corpus_.read_phrases(tok_k_ref_ids);
335
336 // Copy all phrases into a final *sorted* merge set.
337 std::set<SearchResult::Item> final_phrases;
338 // phrases from references
339 for (size_t i = 0; i != top_k_refs.size(); i++) {
340 final_phrases.insert(SearchResult::Item(top_k_refs[i].query, ref_phrases[i]));
341 }
342 // phrases from the raw result set
343 for (const auto& phrase_item : raw_result.phrases()) {
344 const auto& query = phrase_item.query;
345 for (const auto& phrase : phrase_item.result->phrases()) {
346 final_phrases.insert(SearchResult::Item(query, phrase));
347 }
348 }
349
350 // Copy the final top k phrases into the search result.
351 top_k = std::min(final_phrases.size(), max_phrase_count);
352 std::copy_n(final_phrases.begin(), top_k, std::back_inserter(search_result->phrases()));
353
354 return search_result;
355}

363std::string norm_query_to_key(const NormQuery& query) {
364 std::string out;
365 auto it = query.units().begin();
366 auto end = query.units().end();
367 if (it != end) {
368 if (it->tag() == NormQuery::Unit::Tag::WORD) {
369 out.append(*it->text());
370 } else {
371 out.push_back('?');
372 }
373 it++;
374 }
375 for (; it != end; it++) {
376 out.push_back(' ');
377 if (it->tag() == NormQuery::Unit::Tag::WORD) {
378 out.append(*it->text());
379 } else {
380 out.push_back('?');
381 }
382 }
383 return out;
384}

393std::shared_ptr<const RawRefResult> prune(const RawRefResult& result, const SearchOptions& options) {
394 auto pruned = std::make_shared<RawRefResult>();
395
396 // Copy refs
397 if (options.max_phrase_count > 0) {
398 for (const auto& ref : result.refs()) {
399 if (ref.freq() <= options.max_phrase_frequency) {
400 pruned->refs().push_back(ref);
401 if (pruned->refs().size() >= options.max_phrase_count) {
402 break;
403 }
404 }
405 }
406 }
407
408 // Copy unknown words
409 util::vec_append(pruned->unknown_words(), result.unknown_words());
410
411 return pruned;
412}

414std::shared_ptr<const RawRefResult> Netspeak::process_wildcard_query_(const SearchOptions& options,
415 const NormQuery& query) {
416 const auto query_key = norm_query_to_key(query);
417 const auto cached_result = result_cache_.find(query_key);
418
419 if (cached_result && cached_result->options == options) {
420 // exact cache hit
421 return cached_result->result;
422 } else if (cached_result && is_prunable_from(cached_result->options, options)) {
423 // TODO: This very loose compatibility check may result in pathetically
424 // small result sets.
425 return prune(*cached_result->result, options);
426 } else {
427 // can't serve from cache
428 auto final_result = query_processor_.process(options, query);
429 if (cached_result && !final_result->disjoint_with(*cached_result->result)) {
430 // extend the cached phrase refences
431
432 // TODO: These cached reference lists may grow to enormous sizes. Let's
433 // assume that someone wanted to get all results for the query `the ? ?`
434 // and it's easy to see that the cached result set may be gigabytes in
435 // size.
436 auto merged = final_result->merge(*cached_result->result);
437 auto item = std::make_shared<result_cache_item>(options, merged);
438 result_cache_.update(query_key, item);
439 } else {
440 // add this to the cache
441 auto item = std::make_shared<result_cache_item>(options, final_result);
442 result_cache_.insert(query_key, item);
443 }
444 return final_result;
445 }

448std::shared_ptr<const RawPhraseResult> Netspeak::process_non_wildcard_query_(const SearchOptions& options,
449 const NormQuery& query) {
450 auto result = std::make_shared<RawPhraseResult>();
451
452 PhraseDictionary::Value freq_id_pair;
453 if (options.max_phrase_count > 0 && phrase_dictionary_->Get(norm_query_to_key(query), freq_id_pair)) {
454 // found the phrase
455 uint64_t freq = freq_id_pair.e1();
456 if (freq <= options.max_phrase_frequency) {
457 Phrase::Id id(query.size(), freq_id_pair.e2());
458 Phrase phrase(id, freq);
459 for (const auto& unit : query.units()) {
460 phrase.words().push_back(*unit.text());
461 }
462 result->phrases().push_back(std::move(phrase));
463 }
464 } else {
465 // unknown phrase or no phrases requested (max_phrase_count == 0)
466 // in this case, the query may contain unknown words that have to be
467 // reported
468 // TODO: Use the phrase corpus for that
469 for (const auto& unit : query.units()) {
470 const auto& text = *unit.text();
471 if (!phrase_corpus_.contains(text)) {
472 result->unknown_words().push_back(text);
473 }
474 }
475 }
476 return result;
477}

479std::unique_ptr<RawResult> Netspeak::search_raw_(const QueryNormalizer::Options& normalizer_options,
480 const SearchOptions& options, std::shared_ptr<Query> query) {
481 // create norm queries
482 std::vector<NormQuery> normQueries;
483 query_normalizer_.normalize(query, normalizer_options, normQueries);
484
485 // process the norm queries
486 auto result = std::make_unique<RawResult>();
487 for (const auto& query : normQueries) {
488 if (query.has_qmarks()) {
489 result->add_item(query, process_wildcard_query_(options, query));
490 } else {
491 result->add_item(query, process_non_wildcard_query_(options, query));
492 }
493 }
494
495 return result;
496}
