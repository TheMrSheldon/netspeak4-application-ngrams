55  static uint64_t SerializedSizeInBytes(const fs::path& dir) {
56  uint64_t size = 0;
57  if (fs::exists(dir)) {
58  const fs::directory_iterator end;
59  for (fs::directory_iterator it(dir); it != end; ++it) {
60  if (!fs::is_regular_file(it->path()))
61  continue;
62  if (boost::ends_with(it->path().string(), "dat") || boost::ends_with(it->path().string(), "mph")) {
63  size += fs::file_size(it->path());
64  }
65  }
66  }
67  return size;
68  }

78  static BigHashMap* Open(const fs::path& dir, util::memory_type memory = util::memory_type::min_required) {
79  const fs::path idx_file = dir / index_file_name;
80  std::ifstream ifs(idx_file);
81  if (!ifs) {
82  util::throw_runtime_error("Cannot open", idx_file);
83  }
84  std::string part_idx_file;
85  std::vector<Map*> part_maps;
86  const uint64_t num_bytes = SerializedSizeInBytes(idx_file.parent_path());
87  if (num_bytes > util::to_bytes(memory)) {
88  while (std::getline(ifs, part_idx_file)) {
89  part_maps.push_back(ExternalMap<ValueT, IsThreadSafe, Traits>::Open(idx_file.parent_path() / part_idx_file));
90  }
91  } else {
92  while (std::getline(ifs, part_idx_file)) {
93  part_maps.push_back(InternalMap<ValueT, Traits>::Open(idx_file.parent_path() / part_idx_file));
94  }
95  }
96  ifs.close();
97  return new BigHashMap(part_maps);
98  }
