104uint64_t BuildPhraseCorpus(const fs::path& phrase_dir, const fs::path& phrase_corpus_dir) {
105 const fs::path pc_txt_dir = phrase_corpus_dir / PhraseCorpus::txt_dir;
106 const fs::path pc_bin_dir = phrase_corpus_dir / PhraseCorpus::bin_dir;
107 util::check(fs::create_directory(pc_txt_dir), error_message::cannot_create, pc_txt_dir);
108 util::check(fs::create_directory(pc_bin_dir), error_message::cannot_create, pc_bin_dir);
109
110 typedef std::shared_ptr<std::ostream> ostream_pointer;
111 std::unordered_map<size_t, ostream_pointer> phrase_len_to_txt_os;
112 std::unordered_map<size_t, ostream_pointer> phrase_len_to_bin_os;
113 std::unordered_map<size_t, Phrase::Id::Local> phrase_len_to_id;
114 std::unordered_map<std::string, Phrase::Id::Local> unigram_to_id;
115
116 PhraseFileParserItem parser_item;
117 Phrase::Id::Local phrase_id;
118 Phrase::Frequency phrase_freq;
119 const fs::directory_iterator dir_end;
120 for (fs::directory_iterator it(phrase_dir); it != dir_end; ++it) {
121 std::ifstream ifs(it->path());
122 util::check(ifs.is_open(), error_message::cannot_open, it->path());
123 PhraseFileParser<false> parser(ifs);
124 util::log("Processing", it->path());
125 while (parser.read_next(parser_item)) {
126 auto length = parser_item.words.size();
127 // Add words to the vocabulary (assigning a new id to new words).
128 for (const auto& word : parser_item.words) {
129 unigram_to_id.insert(std::make_pair(word, unigram_to_id.size()));
130 }
131 // Create text and binary file if not present.
132 if (phrase_len_to_id.find(length) == phrase_len_to_id.end()) {
133 std::ostringstream oss;
134 oss << PhraseCorpus::phrase_file << '.' << length;
135 const fs::path txt_file = pc_txt_dir / oss.str();
136 phrase_len_to_txt_os[length] = std::shared_ptr<std::ostream>(new std::ofstream(txt_file));
137 oss.clear();
138 oss.str("");
139 oss << PhraseCorpus::phrase_file << '.' << length;
140 const fs::path bin_file = pc_bin_dir / oss.str();
141 phrase_len_to_bin_os[length] =
142 std::shared_ptr<std::ostream>(new std::ofstream(bin_file, std::ios_base::binary));
143 }
144 // Set phrase-id and auto-increment id.
145 parser_item.id = phrase_len_to_id[length]++;
146 // Write phrase in text representation.
147 auto& txt_os = phrase_len_to_txt_os[length];
148 PhraseFileParser<true>::write(*txt_os, parser_item);
149 // Write phrase in binary representation.
150 phrase_freq = parser_item.freq;
151 auto& bin_os = *phrase_len_to_bin_os[length];
152 bin_os.write(reinterpret_cast<const char*>(&phrase_freq), sizeof(Phrase::Frequency));
153 for (const auto& word : parser_item.words) {
154 phrase_id = unigram_to_id[word];
155 bin_os.write(reinterpret_cast<const char*>(&phrase_id), sizeof(Phrase::Id::Local));
156 }
157 }
158 }
159 // Write vocabulary.
160 const fs::path vocab_file = pc_bin_dir / PhraseCorpus::vocab_file;
161 std::ofstream ofs(vocab_file);
162 util::check(ofs.is_open(), error_message::cannot_create, vocab_file);
163 for (const auto& entry : unigram_to_id) {
164 ofs << entry.first << '\t' << entry.second << '\n';
165 }
166 // Calculate expected number of records (need for indexing)
167 uint64_t record_count = 0;
168 for (const auto& entry : phrase_len_to_id) {
169 record_count += entry.first * entry.second;
170 }
171 return record_count;
172}

193void BuildPostlistIndex(const fs::path& phrase_index_dir, const fs::path& postlist_index_dir) {
194 // Load the n-gram index.
195 ai::Configuration config;
196 config.set_index_directory(phrase_index_dir.string());
197 config.set_max_memory_usage(util::memory_type::min_required);
198 ai::Searcher<PhraseIndexValue> searcher(config);
199
200 // Setup postlist index.
201 config.set_index_directory(postlist_index_dir.string());
202 config.set_key_sorting(ai::key_sorting_type::sorted);
203 config.set_value_sorting(ai::value_sorting_type::disabled);
204 ai::Indexer<PostlistIndexValue> indexer(config);
205
206 // Lookup each key and index the returned postlist.
207 const fs::path table_txt = phrase_index_dir / "table.txt";
208 std::ifstream ifs(table_txt);
209 util::check(ifs.is_open(), error_message::cannot_open, table_txt);
210 ai::Record<PostlistIndexValue> postlist_index_record;
211 std::string key, fileid, offset;
212 while (ifs >> key >> fileid >> offset) {
213 const auto postlist = searcher.search_postlist(key);
214 util::check(postlist.get(), "unknown key", key);
215 postlist_index_record.set_key(key);
216 // Postlists shorter than 1000 will not be indexed.
217 const auto postlist_index = IndexPostlist(*postlist, postlist->size() / 1000);
218 for (const auto& entry : postlist_index) {
219 postlist_index_record.set_value(entry);
220 indexer.insert(postlist_index_record);
221 }
222 }
223 indexer.index();
224}

226std::vector<PostlistIndexValue> IndexPostlist(const ai::Postlist<PostlistIndexValue>& postlist,
227 std::size_t resolution) {
228 std::vector<PostlistIndexValue> postlist_index(resolution);
229 if (postlist_index.empty()) {
230 return postlist_index;
231 }
232
233 // Compute normalized total frequency for 1.0 quantile.
234 PostlistIndexValue value;
235 double total_norm_freq = 0;
236 while (postlist.next(value)) {
237 total_norm_freq += static_cast<double>(value.e1()) / std::numeric_limits<PhraseIndexValue::e1_type>::max();
238 }
239
240 // Compute partial normalized frequencies at quantiles.
241 const double step_width = 1.0 / resolution;
242 std::vector<double> norm_freq_at_quantiles(resolution);
243 for (unsigned i = 0; i != norm_freq_at_quantiles.size(); ++i) {
244 norm_freq_at_quantiles[i] = (i + 1) * step_width * total_norm_freq;
245 }
246
247 // Index postlist.
248 postlist.rewind();
249 double norm_freq = 0;
250 size_t quantile_index = 0;
251 double cur_total_norm_freq = 0;
252 PostlistIndexValue index_value;
253 for (unsigned i = 0; postlist.next(value); ++i) {
254 norm_freq = static_cast<double>(value.e1()) / std::numeric_limits<PhraseIndexValue::e1_type>::max();
255 cur_total_norm_freq += norm_freq;
256 assert(cur_total_norm_freq <= total_norm_freq);
257 while (cur_total_norm_freq >= norm_freq_at_quantiles.at(quantile_index)) {
258 // Set current postlist index and value frequency.
259 postlist_index.at(quantile_index++).set(i, value.e1());
260 if (quantile_index == postlist_index.size())
261 break;
262 }
263 if (quantile_index == postlist_index.size())
264 break;
265 }
266 return postlist_index;
267}

288std::vector<std::string> find_missing_words(const std::filesystem::path& phrase_corpus_dir,
289 const std::vector<WordFreqPair>& word_freq_pairs) {

291 for (const auto& pair : word_freq_pairs) {

311void add_missing(std::vector<WordFreqPair>& word_freq_pairs, const std::filesystem::path& phrase_corpus_dir,
312 const Configuration& config) {
313 // read "phrase-corpus/bin/vocab" file to find missing words
314 const auto missing = find_missing_words(phrase_corpus_dir, word_freq_pairs);
315 if (!missing.empty()) {
316 // since the index is basically ready, we will just query "* <word> *" for
317 // every missing word and use the highest frequency.
318 Netspeak netspeak;
319 netspeak.initialize(config);
320
321 for (const auto& word : missing) {
322 service::SearchRequest req;
323 req.set_query("* " + word + " *");
324 req.set_max_phrases(2);
325 req.mutable_phrase_constraints()->set_words_min(2); // it's not a 1-gram
326
327 uint64_t freq = 0;
328
329 service::SearchResponse res;
330 netspeak.search(req, res);
331 if (res.has_result()) {
332 const auto& result = res.result();
333
334 if (result.phrases_size() > 0) {
335 // found at least one phrase with that word
336 freq = result.phrases(0).frequency();
337 } else {
338 // no phrases with that word
339 std::printf("Warning: There are no phrases that with the word \"%s\".", word.c_str());
340 }
341 } else {
342 const auto& error = res.error();
343
344 // error occurred
345 std::printf(
346 "Warning: Unknown error occurred while searching for \"%s\": "
347 "%s : %s\n",
348 word.c_str(), service::SearchResponse::Error::Kind_Name(error.kind()).c_str(), error.message().c_str());
349 }
350
351 std::cout << "Assumed frequency " << freq << " for missing word: " << word << std::endl;
352 word_freq_pairs.push_back(WordFreqPair(word, freq));
353 }
354 }

357void BuildRegexVocabulary(const std::filesystem::path& regex_vocabulary_dir,
358 const std::filesystem::path& phrase_corpus_dir, const Configuration& config) {
359 auto word_freq_pairs = read_1grams(phrase_corpus_dir);
360
361 // Simply going though all 1-grams is not enough because there might be some
362 // words in higher n-grams that are not a 1-gram. There are some reason for
363 // why this might happen but the two most simple are incorrect data sets or
364 // incorrect preprocessing.
365 add_missing(word_freq_pairs, phrase_corpus_dir, config);
366
367 std::sort(word_freq_pairs.begin(), word_freq_pairs.end(), [](const WordFreqPair& a, const WordFreqPair& b) {
368 // sort by frequency
369 return a.second > b.second;
370 });
371
372 // output to file
373 const fs::path regex_vocabulary_file = regex_vocabulary_dir / "vocab.sorted";
374 std::ofstream ofs(regex_vocabulary_file);
375 util::check(ofs.is_open(), error_message::cannot_create, regex_vocabulary_file);
376 for (const auto& entry : word_freq_pairs) {
377 ofs << entry.first << '\n';
378 }
379}

381void MergeDuplicates(const fs::path& phrase_src_dir, const fs::path& phrase_dst_dir) {
382 const char separator = '\t';
383 const std::string tmp_file_prefix = "not_unique.";
384
385 // Create phrase_dst_dir_ph if does not exists
386 if (!fs::exists(phrase_dst_dir)) {
387 fs::create_directory(phrase_dst_dir);
388 }
389
390 // Create partition files in tmp directory.
391 const fs::path tmp_dir = phrase_dst_dir / "tmp";
392 util::check(fs::create_directory(tmp_dir), __func__, error_message::cannot_create, tmp_dir);
393 std::vector<std::shared_ptr<std::ofstream> > tmp_files(
394 util::next_prime(std::distance(fs::directory_iterator(phrase_src_dir), fs::directory_iterator())));
395 for (unsigned i = 0; i != tmp_files.size(); ++i) {
396 const fs::path ph = tmp_dir / (tmp_file_prefix + util::to_string(i));
397 tmp_files[i].reset(new std::ofstream(ph));
398 util::check(tmp_files[i]->is_open(), __func__, error_message::cannot_create, ph);
399 }
400 // Partition all n-grams (equal n-grams go to the same tmp file).
401 std::string line;
402 std::string::size_type tabpos;
403 const fs::directory_iterator dir_end;
404 for (fs::directory_iterator it(phrase_src_dir); it != dir_end; ++it) {
405 std::ifstream ifs(it->path());
406 util::check(ifs.is_open(), __func__, error_message::cannot_open, it->path());
407 util::log("Processing", it->path());
408 while (std::getline(ifs, line)) {
409 tabpos = line.find(separator);
410 if (tabpos == std::string::npos)
411 continue;
412 const size_t hash = util::hash32(line.substr(0, tabpos));
413 *tmp_files[hash % tmp_files.size()] << line << '\n';
414 }
415 }
416 tmp_files.clear(); // Flushes and closes all tmp files.
417
418 // Merge n-grams of each tmp file.
419 std::unordered_map<std::string, uint64_t> ngram_to_freq;
420 for (fs::directory_iterator dir_it(tmp_dir); dir_it != dir_end; ++dir_it) {
421 std::ifstream ifs(dir_it->path());
422 util::check(ifs.is_open(), __func__, error_message::cannot_open, dir_it->path());
423 util::log("Processing", dir_it->path());
424 while (std::getline(ifs, line)) {
425 tabpos = line.find(separator);
426 if (tabpos == std::string::npos) {
427 continue;
428 }
429 const uint64_t phrase_freq = std::stoull(line.substr(tabpos));
430 const auto ret = ngram_to_freq.insert(std::make_pair(line.substr(0, tabpos), phrase_freq));
431 if (!ret.second) {
432 ret.first->second += phrase_freq;
433 }
434 }
435 ifs.close();
436 fs::remove(dir_it->path());
437
438 // Write file with unique n-grams.
439 const fs::path ph = phrase_dst_dir / dir_it->path().filename().string().substr(4); // trim "not_" prefix
440 std::ofstream ofs(ph);
441 util::check(ofs.is_open(), __func__, error_message::cannot_create, ph);
442 for (const auto& entry : ngram_to_freq) {
443 ofs << entry.first << separator << entry.second << '\n';
444 }
445 ngram_to_freq.clear();
446 }
447 fs::remove_all(tmp_dir);
448}
