391 // Star optimization
392 //
393 // The basic idea here is that the star operator has the power to "absorb"
394 // other units. I.e. the star in `the * [ foo ]` can absorb the optional
395 // word "foo" meaning that this query is equivalent to `the *`.
396 //
397 // More generally, the star can absorb any adjacent unit with a minimum
398 // length of 0 (e.g. optional word, other stars). The limiting factor then
399 // becomes the "adjacent" condition but we can switch the order of stars and
400 // QMarks, so we can potentially absorb even more units. E.g. `* ? [ foo ]`
401 // == `? * [ foo ]` == `? *` == `* ?`.
402 //
403 // Note: The relative order of child units is not actually changed. This is
404 // important because the source backtracking can become very confusing
405 // otherwise (with units unexpectedly changing positions).
406
407 // The whole thing is implemented in 2 passes, so we cache the min length
408 std::vector<uint32_t> min_lengths;
409 for (const auto& child : concat.children()) {
410 min_lengths.push_back(child.min_length());
411 }

421 // change the index and check the bounds of the vector
422 if (forward) {
423 index++;
424 return index < concat.children().size();
425 } else {
426 if (index == 0) {
427 return false;
428 } else {
429 index--;
430 return true;
431 }
432 }
433 };

439 // skip qmarks.

445 // so we'll stop here and search for another star

451 };

457 min_lengths.erase(min_lengths.begin() + index);
458 concat.remove_child(index);
459 }
460

462 };

479 }
