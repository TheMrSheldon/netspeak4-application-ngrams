96template <typename T>
97struct value_traits {
98 typedef typename std::enable_if<std::is_arithmetic<T>::value, T>::type value_type;
99 typedef uint16_t io_size_type; // not needed for types with constant size
100
101 static constexpr std::size_t ByteSize() {
102 return sizeof(T);
103 }
104
105 static inline size_t size_of(const value_type& value) {
106 return sizeof(value);
107 }
108
109 static inline char* copy_to(const value_type& value, char* buffer) {
110 std::memcpy(buffer, &value, sizeof(value));
111 return buffer + sizeof(value);
112 }
113
114 static inline const char* copy_from(value_type& value, const char* buffer) {
115 std::memcpy(&value, buffer, sizeof(value));
116 return buffer + sizeof(value);
117 }
118
119 static inline bool write_to(const value_type& value, FILE* file) {
120 return std::fwrite(&value, sizeof(value), 1, file) == 1;
121 }
122
123 static inline bool read_from(value_type& value, FILE* file) {
124 return std::fread(&value, sizeof(value), 1, file) == 1;
125 }
126
127 static inline void print_to(const value_type& value, std::ostream& os) {
128 print_to_os(value, os);
129 }
130
131 static inline void println_to(const value_type& value, std::ostream& os) {
132 print_to(value, os);
133 os << '\n';
134 }
135
140 static inline void parse_from(value_type& value, std::istream& is) {
141 is >> value;
142 if (is.fail()) {
143 throw std::runtime_error(std::format("Reading next {} from input stream failed.", type_name()));
144 }
145 if (is.peek() == tuple_element_separator)
146 is.get();
147 }
148
149 static inline std::string type_name() {
150 return NameOf<value_type>::value;
151 }
152};
