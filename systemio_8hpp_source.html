45inline uint64_t directory_size(const fs::path& dir) {
46 uint64_t num_bytes = 0;
47 const fs::directory_iterator end;
48 for (fs::directory_iterator it(dir); it != end; ++it) {
49 if (fs::is_regular_file(it->path())) {
50 if (!is_hidden_file(it->path())) {
51 num_bytes += fs::file_size(it->path());
52 }
53 } else if (fs::is_directory(it->path())) {
54 num_bytes += directory_size(it->path());
55 }
56 }
57 return num_bytes;
58}

76inline void fread(void* data, size_t size, size_t count, FILE* fs) {
77 // As it turns out, simply asserting nullptr is not enough. As the `std::fread`
78 // spec clearly states:
79 //
80 // > f size or count is zero, fread returns zero and performs no other
81 // > action.
82 //
83 // Since `data` may be the `data()` value of a vector, we have to take this
84 // into account because `data()` may return nullptr for empty vectors depending
85 // on the compiler.
86 assert(data != nullptr || size == 0 || count == 0);
87 assert(fs != nullptr);
88
89 if (std::fread(data, size, count, fs) != count) {
90 signal_error("std::fread failed");
91 }
92}
