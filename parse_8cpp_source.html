30static void unescape_word(std::string& str) {
31 size_t write_i = 0;
32
33 for (size_t i = 0; i < str.size(); i++) {
34 char c = str[i];
35 if (c == '\\') {
36 size_t next = i + 1;
37 if (next < str.size()) {
38 c = str[next];
39 i = next;
40 }
41 }
42 str[write_i] = c;
43 write_i++;
44 }
45
46 str.erase(write_i);
47}

72 void push_stack(std::shared_ptr<Unit> unit) {
73 auto& last = stack_[stack_.size() - 1];
74 last->add_child(unit);
75 stack_.push_back(unit);
76
77 // Right now, the maximum expected stack size is 6 or 7, but was the
78 // grammar changed to be recursive, the stack size has to be limited. The
79 // reason for the limit is not that this stack might grow too much. Almost
80 // all query opertions (e.g. min/max length, conversion to norm queries) are
81 // implemented using recursion, so the program stack might overflow for too
82 // deeply nested query ASTs. So to be on the safe side, the size of this
83 // stack is limited.
84 if (stack_.size() > MaxStackSize) {
85 throw netspeak::invalid_query_error("The query is too deeply nested");
86 }
87 }

89 void pop_stack() {
90 if (stack_.size() <= 1) {
91 throw netspeak::tracable_logic_error("Some operation either pushed too little or popped too much.");
92 }
93 stack_.pop_back();
94 }

205std::shared_ptr<Query> parse_query(const std::string& query) {
206 if (query.length() > QueryConstructor::MaxQueryLength) {
207 throw netspeak::invalid_query_error("Query too long");
208 }
209
210 ANTLRInputStream input(query);
211 QueryLexer lexer(&input);
212 CommonTokenStream tokens(&lexer);
213 QueryParser parser(&tokens);
214
215 QueryErrorHandler error_listener;
216 parser.removeErrorListeners();
217 parser.addErrorListener(&error_listener);
218
219 QueryConstructor construct;
220 antlr4::tree::ParseTreeWalker::DEFAULT.walk(&construct, parser.query());
221 return construct.query();
222}
