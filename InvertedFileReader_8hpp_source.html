35 virtual bool read(record_type& record) {
36 try {
37 while (is_.good()) {
38 switch (is_.peek()) {
39 case '\r':
40 case std::istream::traits_type::eof():
41 is_.get();
42 break;
43 case '\n':
44 is_.get();
45 ++num_lines_;
46 key_traits::parse_from(key_, is_);
47 if (is_.eof())
48 break;
49 // Otherwise fall through here.
50 default:
51 record.key().assign(key_);
52 value_traits::parse_from(record.value(), is_);
53 return true;
54 }
55 }
56 } catch (std::exception& e) {
57 throw std::runtime_error(std::format("{} Line: ", e.what(), num_lines_));
58 }
59 return false;
60 }

62 virtual std::size_t tell() {

64 }
