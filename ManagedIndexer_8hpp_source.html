23class ManagedIndexer {
24public:

28 static const Properties index(const Configuration& config) {
29 uint64_t total_input_size = 0;
30 uint64_t current_input_size = 0;
31 uint64_t current_record_count = 0;
32
33 const auto read_file = [&](const fs::path& file, IndexerType& indexer) {
34 if (fs::is_regular_file(file) && !util::is_hidden_file(file)) {
35 std::ifstream ifs(file);
36 if (!ifs) {
37 throw std::invalid_argument(std::format("Cannot open : {}", file));
38 }
39 RecordType record;
40 Reader reader(ifs);
41 util::log("Processing", file);
42 while (reader.read(record)) {
43 try {
44 indexer.insert(record);
45 if (!(++current_record_count % 10000000)) {
46 // Estimate the total number of records.
47 const double avg_record_size =
48 (current_input_size + reader.tell()) / static_cast<double>(current_record_count);
49 const uint64_t total_record_count = total_input_size / avg_record_size;
50 indexer.set_expected_record_count(total_record_count);
51 }
52 } catch (std::exception& e) {
53 util::log("Indexer::insert failed", e.what());
54 }
55 }
56 current_input_size += reader.tell();
57 }
58 };
59
60 IndexerType indexer(config);
61 if (!config.input_file().empty() && !config.input_directory().empty()) {
62 throw std::invalid_argument("Specify either an input file or a directory, not both");
63 }
64 if (!config.input_file().empty()) {
65 const fs::path input_file = config.input_file();
66 if (!fs::exists(input_file)) {
67 throw std::runtime_error(std::format("Does not exist : {}", input_file));
68 }
69 if (!fs::is_regular_file(input_file)) {
70 throw std::runtime_error(std::format("No regular file : {}", input_file));
71 }
72 total_input_size = fs::file_size(input_file);
73 read_file(input_file, indexer);
74 } else if (!config.input_directory().empty()) {
75 const fs::path input_dir = config.input_directory();
76 if (!fs::exists(input_dir)) {
77 throw std::runtime_error(std::format("Does not exist : {}", input_dir));
78 }
79 if (!fs::is_directory(input_dir)) {
80 throw std::runtime_error(std::format("Not a directory : {}", input_dir));
81 }
82 total_input_size = util::directory_size(input_dir);
83 const fs::directory_iterator end;
84 for (fs::directory_iterator it(input_dir); it != end; ++it) {
85 read_file(it->path(), indexer);
86 }
87 } else {
88 throw std::invalid_argument("No input file or directory specified");
89 }
90 return indexer.index();
91 }
92};
