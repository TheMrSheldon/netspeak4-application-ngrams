84  // Clears the given string and appends characters read from the given stream.
85  // The appending stops when either '\r', '\n', EOF or tuple_element_separator
86  // occur. Except of tuple_element_separator the terminating characters will
87  // not be extracted from the stream so that the outer parser is still able to
88  // detect line endings.
89  static inline void parse_from(value_type& value, std::istream& is) {
90  value.clear();
91  while (is.good()) {
92  switch (is.peek()) {
93  case '\r':
94  case '\n':
95  case std::istream::traits_type::eof():
96  return;
97  case tuple_element_separator:
98  is.get();
99  return;
100  default:
101  value.push_back(is.get());
102  }
103  }
104  }
