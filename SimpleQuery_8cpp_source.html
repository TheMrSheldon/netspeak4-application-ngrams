167bool operator==(const Unit& lhs, const Unit& rhs) {
168 if (lhs.tag() != rhs.tag()) {
169 return false;
170 }
171 if (lhs.is_terminal()) {
172 return lhs.text() == rhs.text();
173 } else {
174 const auto& l_children = lhs.children();
175 const auto& r_children = rhs.children();
176 if (l_children.size() != r_children.size()) {
177 return false;
178 }
179 for (size_t i = 0; i < l_children.size(); i++) {
180 const auto& l_child = l_children[i];
181 const auto& r_child = r_children[i];
182 if (l_child != r_child) {
183 return false;
184 }
185 }
186 return true;
