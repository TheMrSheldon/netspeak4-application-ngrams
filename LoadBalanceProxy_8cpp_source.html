64uint64_t bit_mix(uint64_t x) {
65 // do 16 iterations of xorshift* using x as the seed
66 // https://en.wikipedia.org/wiki/Xorshift#xorshift*
67 for (int i = 16; i > 0; i--) {
68 x ^= x >> 12; // a
69 x ^= x << 25; // b
70 x ^= x >> 27; // c
71 }
72 return x * UINT64_C(0x2545F4914F6CDD1D);
73}

75grpc::Status LoadBalanceProxy::Search_(grpc::ServerContext*, const SearchRequest* request,
76 SearchResponse* response) const {
77 const auto& services_ref = services_;
78
79 auto it = services_ref.find(request->corpus());
80 // check the corpus
81 if (it == services_ref.end()) {
82 auto error = response->mutable_error();
83 error->set_kind(SearchResponse::Error::INVALID_CORPUS);
84 error->set_message("Unknown corpus");
85 return grpc::Status::OK;
86 }
87
88 const auto& services = it->second;
89
90 grpc::ClientContext context;
91 if (services.size() == 1) {
92 // There's only one service available, so just forward the request.
93 return services[0]->Search(&context, *request, response);
94 } else {
95 // There are at least two services, so we have to choose which one to
96 // forward to. To be more cache friendly, the service will be chosen based
97 // on the hash of the query.
98 // The hash will then be processed further to prevent bad hashes. This extra
99 // processing is necessary as an attacker could exploit weaknesses in the
100 // hash function provided by C++ to redirect a huge workload to one service.
101
102 const auto hash = std::hash<std::string>{}(request->query());
103 const auto index = bit_mix(hash) % services.size();
104 return services[index]->Search(&context, *request, response);
105 }
106}
