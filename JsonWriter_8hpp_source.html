20template <class T>
21static void print_json_string(T& out, const std::string& str) {
22 out << '"';
23 for (const auto c : str) {
24 switch (c) {
25 case '\r':
26 out << "\\r";
27 break;
28 case '\n':
29 out << "\\n";
30 break;
31 case '\t':
32 out << "\\t";
33 break;
34 case '\\':
35 out << "\\\\";
36 break;
37 case '"':
38 out << "\\\"";
39 break;
40 default: {
41 const auto unsigned_c = static_cast<unsigned char>(c);
42 if (unsigned_c < ' ') {
43 out << "\\u00" << to_hex_digit(c >> 4) << to_hex_digit(c & 0x0F);
44 } else {
45 out << c;
46 }
47 }
48 }
49 }
50 out << '"';
51}
