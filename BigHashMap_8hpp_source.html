58 static uint64_t SerializedSizeInBytes(const fs::path& dir) {
59 uint64_t size = 0;
60 if (fs::exists(dir)) {
62 const fs::directory_iterator end;
63 for (fs::directory_iterator it(dir); it != end; ++it) {
64 if (!fs::is_regular_file(it->path()))
65 continue;
66 if (boost::ends_with(it->path().string(), "dat") || boost::ends_with(it->path().string(), "mph")) {
67 size += fs::file_size(it->path());
68 }
69 }
70 }
71 return size;

82 static BigHashMap* Open(const fs::path& dir, util::memory_type memory = util::memory_type::min_required) {
83 const fs::path idx_file = dir / index_file_name;
84 std::ifstream ifs(idx_file);
85 if (!ifs) {
86 throw std::runtime_error(std::format("Cannot open : {}", idx_file));
87 }
88 std::string part_idx_file;
89 std::vector<Map*> part_maps;
90 const uint64_t num_bytes = SerializedSizeInBytes(idx_file.parent_path());
91 if (num_bytes > util::to_bytes(memory)) {
92 while (std::getline(ifs, part_idx_file)) {
93 part_maps.push_back(ExternalMap<ValueT, IsThreadSafe, Traits>::Open(idx_file.parent_path() / part_idx_file));
94 }
95 } else {
96 while (std::getline(ifs, part_idx_file)) {
97 part_maps.push_back(InternalMap<ValueT, Traits>::Open(idx_file.parent_path() / part_idx_file));
98 }
99 }
100 ifs.close();
101 return new BigHashMap(part_maps);
102 }
