94 std::vector<uint32_t> id_map;
95
96 const uint32_t len = static_cast<uint32_t>(sorted_words_.size());

98 size_t index = static_cast<size_t>(sorted_words_[i].second);
99 if (index < id_map.size()) {
100 id_map[index] = i;

104 id_map.push_back(len);

108 }
