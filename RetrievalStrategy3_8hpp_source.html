115  void initialize_query(const SearchOptions& options, const NormQuery& query, std::vector<unit_metadata>& metadata) {
116  PhraseDictionary::Value freq_id_pair;
117  for (size_t i = 0; i != query.size(); ++i) {
118  const auto& unit = query.units()[i];
119  if (unit.tag() == NormQuery::Unit::Tag::WORD) {
120  // Note that words not being in the ngram dictionary can still
121  // have a postlist in the ngram index, so this no reason to
122  // skip the entire query.
123  metadata.push_back(unit_metadata());
124  auto& meta = metadata[metadata.size() - 1];
125  meta.position = i;
126 
127  // TODO: This operation is extremely costly and should be replaced by an
128  // in-memory map of all words
129  if (phrase_dictionary_->Get(*unit.text(), freq_id_pair)) {
130  meta.frequency = freq_id_pair.e1();
131  }
132  // TODO: Better stopword detection
133  if (meta.frequency > 1000000000u) { // is stopword
134  meta.pruning = options.pruning_high;
135  } else {
136  meta.pruning = options.pruning_low;
137  }
138  }

142  template <typename OutputIterator>
143  const stats_type initialize_result_set(const unit_metadata& meta, const NormQuery& query,
144  uint64_t max_phrase_frequency, uint64_t max_phrase_count,
145  OutputIterator output) {
146  max_phrase_frequency = std::min(max_phrase_frequency, compute_jumpin_frequency_(query));
147 
148  stats_type stats;
149  std::shared_ptr<invertedindex::Postlist<index_entry_type> > postlist(
150  search_(make_key(query, meta), max_phrase_frequency, meta.pruning));
151  if (!postlist) {
152  stats.unknown_word = *(query.units()[meta.position].text());
153  return stats;
154  }
155 
156  // The first of the following two _while_ loops has two tasks:
157  // (1) Skip values of the postlist with invalid frequency
158  // (2) Select the frequency of the first valid _index_entry_
159 
160  index_entry_type index_entry;
161  while (postlist->next(index_entry) && max_phrase_count != 0) {
162  // Depending on the resolution of the postlist index,
163  // search_() can only roughly satisfy the _max_freq_ condition,
164  // so we have to check this condition here again.
165  ++stats.eval_index_entry_count;
166  if (traits::get_phrase_frequency(index_entry) <= max_phrase_frequency) {
167  --max_phrase_count;
168  stats.max_phrase_frequency = traits::get_phrase_frequency(index_entry);
169  *output = index_entry;
170  ++output;
171  break;
172  }
173  }
174  // Copy all remaining index entries.
175  while (postlist->next(index_entry) && max_phrase_count != 0) {
176  --max_phrase_count;
177  ++stats.eval_index_entry_count;
178  *output = index_entry;
179  ++output;
180  }
181  stats.min_phrase_frequency = traits::get_phrase_frequency(index_entry);
182  return stats;
183  }

264  static std::vector<std::string> extract_longest_substrings(const NormQuery& query) {
265  size_t max_word_count = 1;
266  std::vector<std::string> substrings;
267  std::vector<std::string> word_buffer;
268  for (const auto& unit : query.units()) {
269  if (unit.tag() == NormQuery::Unit::Tag::WORD) {
270  // TODO: Can the empty case every happen?
271  if (!unit.text()->empty()) {
272  word_buffer.push_back(*unit.text());
273  }
274  } else {
275  if (word_buffer.size() >= max_word_count) {
276  const auto substring = boost::join(word_buffer, " ");
277  if (word_buffer.size() > max_word_count) {
278  max_word_count = word_buffer.size();
279  substrings.clear();
280  }
281  substrings.push_back(substring);
282  }
283  word_buffer.clear();
284  }
285  }
286  if (word_buffer.size() >= max_word_count) {
287  const auto substring = boost::join(word_buffer, " ");
288  if (word_buffer.size() > max_word_count) {
289  substrings.clear();
290  }
291  substrings.push_back(substring);
292  }
293  return substrings;
294  }

296  uint64_t compute_jumpin_frequency_(const NormQuery& query) {
297  PhraseDictionary::Value freq_id_pair;
298  uint64_t min_frequency = std::numeric_limits<uint64_t>::max();
299  for (const auto& substring : extract_longest_substrings(query)) {
300  // Note that substrings (n-grams) not being in the ngram dictionary
301  // can still be contained in the ngram index as (n+x)-grams.
302  // Therefore this is no reason to skip the entire query.
303  if (phrase_dictionary_->Get(substring, freq_id_pair)) {
304  min_frequency = std::min(min_frequency, freq_id_pair.e1());
305  }
306  }
307  return min_frequency;
308  }

310  std::shared_ptr<invertedindex::Postlist<index_entry_type> > search_(const std::string& key, size_t max_freq,
311  uint32_t pruning) {
312  // Get index_begin from max_freq (jumpin frequency).
313  size_t idx_begin = 0;
314  postlist_index_value_type prev_index_value;
315  postlist_index_value_type cur_index_value;
316  std::unique_ptr<invertedindex::Postlist<postlist_index_value_type> > meta_postlist(
317  postlist_index_.search_postlist(key));
318  if (meta_postlist) {
319  while (meta_postlist->next(cur_index_value)) {
320  if (cur_index_value.e2() == max_freq) {
321  idx_begin = cur_index_value.e1();
322  break;
323  }
324  if (cur_index_value.e2() < max_freq) {
325  // choose the value before, otherwise
326  // entries will be lost in some cases
327  idx_begin = prev_index_value.e1();
328  break;
329  }
330  prev_index_value = cur_index_value;
331  }
332  }
333 
334  // Read postlist from ngram index.
335  return std::shared_ptr<invertedindex::Postlist<index_entry_type> >(
336  phrase_index_.search_postlist(key, idx_begin, pruning));
337  }
